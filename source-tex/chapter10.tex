\chapter{העתקות לינאריות}\label{ch:linearmaps}

בחשבון דיפרנציאלי ואינטגרלי עוסקים בהרחבה בפונקציות של משתנה ממשי (בהמשך עוברים למספר משתנים). באופן יותר כללי במתמטיקה, פונקציה מתארת חוק שמקשר בין קלט לפלט: לכל אובייקט במרחב אחד מותאם אובייקט במרחב אחר (אובייקטים אלה אינם בהכרח מספרים). פונקציות יכולות להיות פשוטות או מורכבות מאוד, ולעיתים קשה להבין את פעולתן על כל הקלטים האפשריים.

באלגברה לינארית מתמקדים במחלקה מיוחדת של פונקציות – העתקות לינאריות. אלו פונקציות בין מרחבים וקטוריים השומרות על חיבור וכפל בסקלר (בדומה לתכונות של כפל מטריצה בוקטור), ולכן ההתנהגות שלהן נקבעת לחלוטין על ידי פעולתן על וקטורי בסיס כלשהו למרחב המקורי (תחום ההעתקה). תכונה זו מאפשרת תיאור מלא בעזרת מטריצות ונותנת כלים חזקים לניתוח וחישוב. נשים דגש על הקשר בין העתקות לינאריות למטריצות.

בהנדסה, העתקות לינאריות משמשות כמודל בסיסי למערכות רבות, כגון מערכות חשמליות, מערכות מכניות, מערכות בקרה ועיבוד אותות. במערכות כאלה מתקיים בהרבה מקרים עקרון הסופרפוזיציה: השפעתם של כמה קלטים על מערכת היא סכום ההשפעות של כל קלט בנפרד, וכפל קלט בסקלר גורר שכפול זהה של הפלט. לדוגמה, במעגל חשמלי לינארי, אם מקור מתח אחד יוצר זרם מסוים ומקור נוסף יוצר זרם אחר, אז הפעלת שניהם יחד תיצור זרם שהוא סכום הזרמים שכל אחד יוצר לבדו.

המשמעות המתמטית של סופרפוזיציה היא שניתן לתאר את הקשר בין קלט לפלט באמצעות העתקה לינארית. גם כאשר המערכת האמיתית אינה לינארית לחלוטין, מהנדסים משתמשים בהעתקות לינאריות כקירוב מקומי המאפשר ניתוח, חישוב ותכנון יעילים.

\section{הגדרות ותכונות}

נעבור להגדרה המדויקת של פונקציה כהקדמה להגדרה של העתקה לינארית:

\begin{definition}
יהיו קבוצות
$A,B\neq\emptyset$.
פונקציה 
$f$
מקבוצה 
$A$
ל-
$B$
מתאימה לכל איבר 
$a\in A$
בדיוק איבר אחד 
$b\in B$.
מסמנים 
$f:A\to B$
וגם
$b=f(a)$.
$A$
נקראת התחום של 
$f$,
ו-
$B$
נקראת הטווח שלה.
\end{definition}

    \centering
    \includegraphics[width=0.6\textwidth]{images/Function.png}
    \caption{מכל איבר בתחום יוצא בדיוק חץ אחד לאיבר בטווח}
    \label{fig:Function}

\begin{remark}
בהקשרים מסוימים יש מילים נרדפות לפונקציה, כמו "העתקה", "טרנספורמציה" ו"אופרטור". בקורס שלנו נדבר על העתקות, ובפרט העתקות לינאריות. העתקה לינארית היא פונקציה בין מרחבים וקטוריים שמקיימת זוג תכונות שיחד נקראות "לינאריות".
\end{remark}

\begin{definition}
יהיו
$V,W$
מרחבים וקטוריים מעל
$\F$.
פונקציה
$T:V\to W$
נקראת העתקה לינארית (ה"ל) אם לכל
$v_1,v_2\in V$
ולכל
$\alpha\in\F$
מתקיים:
$$
\begin{aligned}
T(v_1+v_2)&=T(v_1)+T(v_2) \\
T(\alpha v_1)&=\alpha T(v_1)
\end{aligned}
$$
\end{definition}

\begin{remark}
כאשר נתייחס לה"ל
$T:V\to W$
ההנחה המובלעת היא ש-
$V,W$
הם מרחבים וקטוריים מעל 
$\F$.
אין משמעות ללינאריות בלי מרחבים וקטוריים.

שימו לב שהחיבור
$v_1+v_2$
מתבצע במרחב 
$V$, 
ואילו החיבור
$T(v_1)+T(v_2)$
מתבצע במרחב 
$W$.
כנ"ל לכפל בסקלר - כל פעולה מתבצעת במרחב הרלוונטי. אז הסימונים קצת מתעתעים, כי זה לא בהכרח אותו סוג חיבור או אותו סוג של כפל בסקלר בשני המרחבים.
\end{remark}

%% הדגמה דומה לזו שהייתה כבר לכפל מטריצה בוקטור

\begin{example}\
\begin{enumerate}[label=\alph*.]
\item 
$f:\R\to\R$
המוגדרת ע"י
$f(x)=2x$
היא ה"ל
(ניתן לחשוב על
$\R$
כעל
$\R^1$,
שהוא מ"ו ממימד 
$1$).
לכל
$x_1,x_2,\alpha\in\R$
מתקיים:
$$
\begin{aligned}
f(x_1+x_2)&=2(x_1+x_2)=2x_1+2x_2=f(x_1)+f(x_2) \\
f(\alpha x)&=2(\alpha x)=\alpha(2x)=\alpha f(x)
\end{aligned}
$$


\item
$T:\R^2\to\R$
המוגדרת ע"י
$T\begin{pmatrix}x \\y\end{pmatrix}=x+y$
היא ה"ל. לכל 
$\begin{pmatrix}x_1 \\y_1\end{pmatrix},\begin{pmatrix}x_2 \\y_2\end{pmatrix}\in\R^2$
ולכל 
$\alpha\in\F$
מתקיים:
$$
\begin{aligned}
T\left(\begin{pmatrix}x_1 \\y_1\end{pmatrix}+\begin{pmatrix}x_2 \\y_2\end{pmatrix}\right)&=T\begin{pmatrix}x_1+x_2 \\y_1+y_2\end{pmatrix}=(x_1+x_2)+(y_1+y_2)\\
&=(x_1+y_1)+(x_2+y_2)=T\begin{pmatrix}x_1 \\y_1\end{pmatrix}+T\begin{pmatrix}x_2 \\y_2\end{pmatrix}\\
T\left(\alpha\begin{pmatrix}x_1 \\y_1\end{pmatrix}\right)&=T\begin{pmatrix}\alpha x_1 \\\alpha y_1\end{pmatrix}=\alpha x_1+\alpha y_1=\alpha(x_1+y_1)=\alpha T\begin{pmatrix}x_1 \\y_1\end{pmatrix}
\end{aligned}
$$


\item 
הפונקציה
$f:\R\to\R$
המוגדרת ע"י
$f(x)=x+1$
אינה ה"ל. בתור דוגמה נגדית, ניקח
$x_1=0,x_2=1,\alpha=2$
ונקבל:
$$
\begin{aligned}
f(x_1+x_2)&=f(1)=2\neq 1+2=f(0)+f(1)=f(x_1)+f(x_2) \\
f(2x_1)&=f(0)=1\neq 2\cdot 1=2f(0)=2f(x_1)
\end{aligned}
$$
הפרכנו את שתי התכונות בהגדרה של ה"ל (מספיק להפריך אחת).


\item 
הפונקציה
$f:\R\to\R$
המוגדרת ע"י
$f(x)=x^2$
אינה ה"ל. בתור דוגמה נגדית, ניקח
$x_1=1,x_2=-1,\alpha=2$
ונקבל:
$$
\begin{aligned}
f(x_1+x_2)&=f(0)=0\neq 1+1=f(1)+f(-1)=f(x_1)+f(x_2) \\
f(2x_1)&=f(2)=4\neq 2\cdot 1=2f(1)=2f(x_1)
\end{aligned}
$$


\item 
העתקת השחלוף
$T:\M_{n\times n}(\F)\to\M_{n\times n}(\F)$
המוגדרת ע"י
$T(A)=A^t$
היא לינארית. הוכחנו זאת בטענה
\ref{prop:transpose},
לפיה לכל 
$A,B\in\M_{n\times n}(\F)$
ולכל 
$\alpha\in\F$
מתקיים
$$
\begin{aligned}
T(A+B)&=(A+B)^t=A^t+B^t=T(A)+T(B) \\
T(\alpha A)&=(\alpha A)^t=\alpha A^t=\alpha T(A)
\end{aligned}
$$


\item 
יהי 
$V$
מ"ו מעל
$\F$
ממימד
$n$
עם בסיס סדור
$B$.
הפונקציה
$T:V\to\F^n$
המוגדרת ע"י
$T(v)=[v]_B$
היא לינארית. הוכחנו זאת בטענה
\ref{prop:coordinates-linear},
לפיה לכל
$v_1,v_1\in V$
ולכל
$\alpha\in\F$
מתקיים:
$$
\begin{aligned}
T(v_1+v_2)&=[v_1+v_2]_B=[v_1]_B+[v_2]_B=T(v_1)+T(v_2) \\
T(\alpha v_1)&=[\alpha v_1]_B=\alpha[v_1]_B=\alpha T(v_1)
\end{aligned}
$$


\end{enumerate}
\end{example}

למעשה, לכל ה"ל
$T:\F\to\F$
יש נוסחה פשוטה, בין אם
$\F=\R$
או 
$\F=\C$.

\begin{proposition}
תהי
$T:\F\to\F$
ה"ל. אז קיים
$c\in\F$
כך ש-
$T(x)=cx$.
\end{proposition}

\begin{proof}
נסמן
$c=T(1)$.
לכל
$x\in\F$
מתקיים
$$
.T(x)=T(x\cdot 1)=xT(1)=x\cdot c=cx
$$
שימו לב שתחילה התייחסנו אל 
$x$
כוקטור ואחר כך כסקלר (הוא שניהם במקרה של 
$\F$).
\end{proof}

נרצה להוכיח תכונות נוספות של ה"ל, שנובעות מהתכונות בהגדרה.
\begin{proposition}\label{prop:more-linearity}
יהיו 
$V,W$
מרחבים וקטוריים מעל
$\F$,
ותהי
$T:V\to W$
ה"ל. אז מתקיים:

\begin{enumerate}[label=\alph*.]
\item 
לכל
$v_1,...,v_k\in V$
ולכל
$\alpha_1,...,\alpha_k\in \F$
מתקיים
$$
.T(\alpha_1 v_1+...+\alpha_k v_k)=\alpha_1 T(v_1)+...+\alpha_k T(v_k)
$$


\item 
$$T(0_V)=0_W$$


\end{enumerate}
\end{proposition}

\begin{proof}\
\begin{enumerate}[label=\alph*.]
\item
יהיו
$v_1,...,v_k\in V$
ו-
$\alpha_1,...,\alpha_k\in \F$.
נשתמש בתכונת החיבור של לינאריות (לשני מחוברים) מספר חוזר של פעמים כדי להוציא כל מחובר מחוץ לסוגריים אחד אחרי השני (ההוכחה הפורמלית היא באינדוקציה למי שמכיר):

$$
\begin{aligned}
&T(\alpha_1 v_1+...+\alpha_{k-1}v_{k-1}+\alpha_k v_k)=T(\alpha_1 v_1+...+\alpha_{k-1}v_{k-1})+T(\alpha_k v_k)\\
&=T(\alpha_1 v_1+...+\alpha_{k-2}v_{k-2})+T(\alpha_{k-1}v_{k-1})+T(\alpha_k v_k)=...\\
&=T(\alpha_1 v_1)+...+T(\alpha_k v_k)=\alpha_1T(v_1)+...+\alpha_k T(v_k)
\end{aligned}
$$
במעבר האחרון הוצאנו סקלר מכל ערך של 
$T$
לחוד.


\item 
מתקיים
$$
.T(0_V)=T(0\cdot 0_V)=0\cdot T(0_V)=0_W
$$


\end{enumerate}
\end{proof}

\begin{exercise}
האם
$T:\R^3\to M_{2\times 2}(\R)$
המוגדרת ע"י
$$
T\begin{pmatrix}x\\y\\z\end{pmatrix}=\begin{pmatrix}
x+1 & y \\
z & 0
\end{pmatrix}
$$
היא ה"ל?
\end{exercise}

\begin{solution}
$T$
אינה לינארית. הדרך הפשוטה ביותר להראות זאת היא לשים לב שמתקיים
$$
.T\begin{pmatrix}0\\0\\0\end{pmatrix}=\begin{pmatrix}
1 & 0 \\
0 & 0
\end{pmatrix}\neq \begin{pmatrix}
0 & 0 \\
0 & 0
\end{pmatrix}
$$
\end{solution}

\begin{remark}
התנאי
$T(0_V)=0_W$
הוא הכרחי ללינאריות, אך הוא אינו מספיק. למשל, ראינו כבר כי הפונקציה
$f:\R\to\R$
המוגדרת ע"י
$f(x)=x^2$
אינה ה"ל, אם כי מתקיים
$f(0)=0$.
\end{remark}

\subsection{העתקה לינארית המוגדרת ע"י מטריצה}
לאור כל מה שלמדנו על מטריצות, נוח לעבוד עם העתקה לינארית המוגדרת ע"י מטריצה.
\begin{definition}
תהי 
$A\in\M_{m\times n}(\F)$.
נגדיר
$T_A:\F^n\to\F^m$
ע"י
$$
.T_A(v)=Av
$$
\end{definition}

\begin{proposition}\label{prop:matrix-map}
לכל 
$A\in\M_{m\times n}(\F)$
ההעתקה
$T_A:\F^n\to\F^m$
היא ה"ל.
\end{proposition}

\begin{proof}
ההוכחה מיידית לפי טענה
\ref{prop:matrix-vector}.
\end{proof}

\begin{example}\
\begin{enumerate}[label=\alph*.]
\item 
עבור
$$
A=\begin{pmatrix}
1 & 2 & -3 \\
2 & 0 & 1
\end{pmatrix}
$$
נקבל 
$T_A:\R^3\to \R^2$
הנתונה ע"י
$$
.T_A\begin{pmatrix}x\\y\\z\end{pmatrix}=\begin{pmatrix}
1 & 2 & -3 \\
2 & 0 & 1
\end{pmatrix}\begin{pmatrix}x\\y\\z\end{pmatrix}=\begin{pmatrix}
x+2y-3z \\
2x+z
\end{pmatrix}
$$


\item
תהי
$\theta\in\R$.
נסמן
$$
.A=\begin{pmatrix}
\cos{\theta} & -\sin{\theta} \\
\sin{\theta} & \cos{\theta} 
\end{pmatrix}
$$
נקבל 
$T_A:\R^2\to \R^2$
הנתונה ע"י
$$
.T_A\begin{pmatrix}x\\y\end{pmatrix}=\begin{pmatrix}
\cos{\theta} & -\sin{\theta} \\
\sin{\theta} & \cos{\theta} 
\end{pmatrix}\begin{pmatrix}x\\y\end{pmatrix}=\begin{pmatrix}
x\cos{\theta}-y\sin{\theta} \\
x\sin{\theta}+y\cos{\theta}
\end{pmatrix}
$$

במקרה זה ל-
$T_A$
יש משמעות גיאומטרית של סיבוב וקטור הקלט
$\begin{pmatrix}x\\y\end{pmatrix}$
בזווית
$\theta$
נגד כיוון השעון. זו פעולה אנלוגית להכפלת
$z=x+iy$
במספר
$e^{i\theta}=\cos{\theta}+i\sin{\theta}$
במישור המרוכב 
$\C$.
אכן:
$$
(\cos{\theta}+i\sin{\theta})(x+iy)=x\cos{\theta}-y\sin{\theta}+i(x\sin{\theta}+y\cos{\theta})
$$
%% הדגמת הסיבוב


\item 
נסמן
$$
.A=\begin{pmatrix}
0 & 1 \\
1 & 0
\end{pmatrix}
$$
נקבל 
$T_A:\R^2\to \R^2$
הנתונה ע"י
$$
.T_A\begin{pmatrix}x\\y\end{pmatrix}=\begin{pmatrix}
0 & 1 \\
1 & 0
\end{pmatrix}\begin{pmatrix}x\\y\end{pmatrix}=\begin{pmatrix}
y \\
x
\end{pmatrix}
$$
כאן המשמעות הגיאומטרית היא שיקוף ביחס לישר 
$y=x$.

%% ציור או הדגמה

\end{enumerate}
\end{example}

\begin{exercise}
הראו כי
$T:\R^2\to\R^3$
הנתונה ע"י
$$
T\begin{pmatrix}x\\y\\z\end{pmatrix}=\begin{pmatrix}2x+5y+z\\3x-y+2z\\y-5z\end{pmatrix}
$$
היא ה"ל.
\end{exercise}

\begin{solution}
נמצא
$A\in\M_{3\times 3}(\R)$
כך ש-
$T=T_A$.
מכך ינבע כי 
$T$
היא ה"ל לפי טענה
\ref{prop:matrix-map}.
המעבר מהנוסחה של
$T$
למטריצה
$A$
אנלוגי לחלוטין למעבר מממ"ל למטריצת המקדמים המצומצמת. לכן, עבור
$$
A=\begin{pmatrix}
2 & 5 & 1\\
3 & -1 & 2\\
0 & 1 & -5
\end{pmatrix}
$$
נקבל
$$
.T_A\begin{pmatrix}x\\y\\z\end{pmatrix}=\begin{pmatrix}
2 & 5 & 1\\
3 & -1 & 2\\
0 & 1 & -5
\end{pmatrix}\begin{pmatrix}x\\y\\z\end{pmatrix}=\begin{pmatrix}2x+5y+z\\3x-y+2z\\y-5z\end{pmatrix}=T\begin{pmatrix}x\\y\\z\end{pmatrix}
$$
זה בדיוק מראה כי
$T=T_A$,
ולכן היא ה"ל.
\end{solution}

\begin{proposition}
תהי
$T:\F^n\to\F^m$
ה"ל. אז קיימת
$A\in\M_{m\times n}(\F)$
כך ש-
$T=T_A$.
\end{proposition}

\begin{proof}
נבחר את 
$A$
להיות המטריצה שעמודותיה הן
$T(e_1),...,T(e_n)$:
$$
A=\begin{pmatrix}
\mid & \mid &        & \mid \\
T(e_1) & T(e_2) & \cdots & T(e_n) \\
\mid & \mid &        & \mid
\end{pmatrix}
$$
לכל 
$\begin{pmatrix}x_1\\\vdots\\x_n\end{pmatrix}\in\F^n$
מתקיים:
$$
\begin{aligned}
T_A\begin{pmatrix}x_1\\\vdots\\x_n\end{pmatrix}&=\begin{pmatrix}
\mid & \mid &        & \mid \\
T(e_1) & T(e_2) & \cdots & T(e_n) \\
\mid & \mid &        & \mid
\end{pmatrix}\begin{pmatrix}x_1\\\vdots\\x_n\end{pmatrix}=x_1T(e_1)+...+x_nT(e_n)\\
&=T(x_1e_1+...x_ne_n)=T\begin{pmatrix}x_1\\\vdots\\x_n\end{pmatrix}
\end{aligned}
$$
קיבלנו כי
$T_A=T$
כנדרש.
\end{proof}

\begin{example}
נתונה ה"ל 
$T:\R^2\to\R^2$
המקיימת
$$
.T\begin{pmatrix}1 \\ 0\end{pmatrix}=\begin{pmatrix}1 \\ 2\end{pmatrix},\quad T\begin{pmatrix}0 \\ 1\end{pmatrix}=\begin{pmatrix}-2 \\ 1\end{pmatrix}
$$

ראשית, נמצא לה נוסחה רגילה (בלי מטריצה) ע"י שימוש בנתונים ובתכונות של ה"ל. לכל
$\begin{pmatrix}x\\y\end{pmatrix}\in\R^2$
מתקיים:
$$
\begin{aligned}
T\begin{pmatrix}x\\y\end{pmatrix}&=T\left(x\begin{pmatrix}1\\0\end{pmatrix}+y\begin{pmatrix}0\\1\end{pmatrix}\right)=xT\begin{pmatrix}1\\0\end{pmatrix}+y\begin{pmatrix}0\\1\end{pmatrix}=x\begin{pmatrix}1 \\ 2\end{pmatrix}+y\begin{pmatrix}-2 \\ 1\end{pmatrix} \\
&=\begin{pmatrix}x-2y \\ 2x+y \end{pmatrix}
\end{aligned}
$$

לבסוף, נשים לב שאפשר להציג את הפעולה של 
$T$
ככפל של מטריצה בוקטור. מתקיים

$$
.T\begin{pmatrix}x\\y\end{pmatrix}=\begin{pmatrix}
1 & -2 \\
2 & 1
\end{pmatrix}\begin{pmatrix}x\\y\end{pmatrix}
$$

מאחר שהקלטים הנתונים הם וקטורי יחידה, מלכתחילה ניתן לכתוב 
$T=T_A$.
אבל טוב גם להבין את הגישה הישירה בלי מטריצה.

\end{example}

\section{קיום ויחידות העתקה לינארית לפי בסיס}\label{section:map-by-basis}

ראינו כי במקרה הפרטי של ה"ל 
$T:\F^n\to\F^m$,
הערכים 
$T(e_1),...,T(e_n)$
קובעים את
$T$
ביחידות - ואף הצלחנו לבטא את הקשר ע"י מטריצה. נרצה להוכיח משפט כללי יותר על ה"ל בין מרחבים וקטוריים שאינם בהכרח מהצורה 
$\F^k$,
וגם עבור ערכים המתאימים לוקטורי בסיס שאינו בהכרח סטנדרטי. המשפט הבא נקרא "משפט ההגדרה":

\begin{theorem}\label{thm:exist-unique}
יהיו 
$V,W$
מרחבים וקטוריים מעל
$\F$,
$B=\Set{v_1,...,v_n}$
בסיס ל-
$V$
ו-
$\Set{w_1,...,w_n}\subseteq W$
תת-קבוצה כלשהי. אז קיימת ויחידה ה"ל 
$T:V\to W$
המקיימת
$$
.T(v_1)=w_1,...,T(v_n)=w_n
$$
\end{theorem}

\begin{proof}
יהי
$v\in V$.
נסמן
$$
.[v]_B=\begin{pmatrix}\alpha_1\\ \vdots \\ \alpha_n\end{pmatrix}
$$
בתור התחלה, נניח שקיימת ה"ל
$T:V\to W$
השולחת כל 
$v_k$
ל-
$w_k$
המתאים. אז לפי טענה 
\ref{prop:more-linearity}
בהכרח מתקיים
$$
.T(v)=T(\sum_{k=1}^n \alpha_k v_k)=\sum_{k=1}^n \alpha_k T(v_k)=\sum_{k=1}^n \alpha_k w_k
$$
אם כך, נפעל בכיוון ההפוך: נגדיר
$$
,T(v)=\sum_{k=1}^n \alpha_k w_k
$$
ונוכיח לינאריות. אם נצליח, הרי ש-
$T$
היא הה"ל היחידה שמקיימת את הדרישות. ובכן, יהי
$w\in V$
ונסמן
$$
.[w]_B=\begin{pmatrix}\beta_1\\ \vdots \\ \beta_n\end{pmatrix}
$$
מתקיים
$$
,[v+w]_B=\begin{pmatrix}\alpha_1+\beta_1\\ \vdots \\ \alpha_n +\beta_n\end{pmatrix}
$$
ומכאן נקבל לפי הגדרת
$T$:
$$
\begin{aligned}
T(v+w)&=T(\sum_{k=1}^n(\alpha_k+\beta_k)v_k)=\sum_{k=1}^n(\alpha_k+\beta_k)w_k \\
&=\sum_{k=1}^n\alpha_kw_k+\sum_{k=1}^n\beta_kw_k=T(v)+T(w)
\end{aligned}
$$
נבדוק גם כפל בסקלר: לכל
$\gamma\in\F$
מתקיים
$$
.T(\gamma v)=T(\sum_{k=1}^n\gamma\alpha_kv_k)=\sum_{k=1}^n\gamma\alpha_kw_k=\gamma\sum_{k=1}^n\alpha_k w_k =\gamma T(v)
$$

\end{proof}

\begin{remark}
נניח כי
$V,W$
מרחבים וקטוריים מעל 
$\F$
כך ש-
$\dim(V)=n$.
אם מחפשים ה"ל
$T:V\to W$
המקיימת
$$
,\forall 1\leq i\leq k \quad T(v_i)=w_i
$$
יש כמה מקרים:

\begin{itemize}
\item 
$S=\Set{v_1,...,v_k}$
היא בסיס ל-
$V$
(בפרט
$k=n$).
במקרה זה, קיימת ה"ל
$T$
יחידה לפי משפט ההגדרה.


\item
$S$
בת"ל, אך
$k<n$.
במקרה זה, 
$T$
קיימת אך אינה יחידה, כי ניתן להשלים את
$S$
לבסיס ל-
$V$
ע"י הוספת וקטורים
$v_{k+1},...,v_n$.
הערכים 
$T(v_{k+1}),...,T(v_n)$
שרירותיים כי אין שום דרך לקבוע אותם לפי הנתונים. כל בחירה של הערכים החדשים היא לגיטימית ולכל אחת מתאימה ה"ל 
$T$,
כך שיש אינסוף אפשרויות ל-
$T$
המקיימות את הנתונים.


\item
$S$
ת"ל. במקרה זה יש לחלץ תת-קבוצה בת"ל בגודל מקסימלי, נניח
$S'=\Set{v_1,...,v_j}$
עבור 
$j<k$.
יש לבדוק כי הערכים
$T(v_{j+1},...,v_n)$
מקיימים את דרישת הלינאריות. כלומר, לכל 
$j<i\leq k$
קיימים 
$a_1,...,a_j\in\F$
כך ש-
$$
,v_i=a_1v_1+...+a_jv_j
$$
ולכן צריך להתקיים
$$
.T(v_i)=a_1T(v_1)+...+a_jT(v_j)
$$
אם זה לא מתקיים אפילו עבור ערך אחד של 
$j<i\leq k$,
אז זו סתירה ללינאריות ולכן לא קיימת ה"ל
$T$
שמקיימת את הנתונים. אחרת, צריך למצוא את
$T$
לפי הנתונים הקשורים ל-
$S'$.
אם 
$j=n$,
אז קיבלנו את המקרה הראשון (שמתאים למשפט). אם 
$j<n$,
אז קיבלנו את המקרה השני (וצריך להשלים לבסיס ולבחור ערכים שרירותיים).


\end{itemize}
\end{remark}

\begin{example}\
\begin{enumerate}[label=\alph*.]
\item
נברר אם קיימת ה"ל
$T:\R^3\to\R^3$
המקיימת
$$
.T\begin{pmatrix}1\\-1\\1\end{pmatrix}=\begin{pmatrix}1\\2\\3\end{pmatrix},\quad T\begin{pmatrix}2\\1\\0\end{pmatrix}=\begin{pmatrix}4\\-1\\0\end{pmatrix},\quad T\begin{pmatrix}4\\-1\\2\end{pmatrix}=\begin{pmatrix}0\\0\\0\end{pmatrix}
$$
נשים לב כי וקטורי הקלט
$\begin{pmatrix}1\\-1\\1\end{pmatrix},\begin{pmatrix}2\\1\\0\end{pmatrix},\begin{pmatrix}4\\-1\\2\end{pmatrix}$
הם ת"ל. יש למצוא את משוואת התלות הלינארית ביניהם ולבדוק שוקטורי הפלט גם מקיימים אותה. דירוג מראה כי
$$
.\begin{pmatrix}4\\-1\\2\end{pmatrix}=2\begin{pmatrix}1\\-1\\1\end{pmatrix}+\begin{pmatrix}2\\1\\0\end{pmatrix}
$$
נניח בשלילה שקיימת ה"ל 
$T$
שמקיימת את הנתונים. אז נובע כי
$$
,T\begin{pmatrix}4\\-1\\2\end{pmatrix}=2T\begin{pmatrix}1\\-1\\1\end{pmatrix}+T\begin{pmatrix}2\\1\\0\end{pmatrix}
$$
ולאחר הצבת הנתונים נקבל
$$
.\begin{pmatrix}0\\0\\0\end{pmatrix}=2\begin{pmatrix}1\\2\\3\end{pmatrix}+\begin{pmatrix}4\\-1\\0\end{pmatrix}
$$
זו סתירה כי אגף ימין שווה ל-
$\begin{pmatrix}6\\3\\6\end{pmatrix}$.
המסקנה היא שלא קיימת ה"ל 
$T$
המקיימת את הנתונים.


\item
נמצא
$T:\R^3\to\R^3$
המקיימת
$$
.T\begin{pmatrix}1\\-1\\1\end{pmatrix}=\begin{pmatrix}1\\2\\3\end{pmatrix},\quad T\begin{pmatrix}2\\1\\0\end{pmatrix}=\begin{pmatrix}4\\-1\\0\end{pmatrix},\quad T\begin{pmatrix}4\\-1\\2\end{pmatrix}=\begin{pmatrix}6\\3\\6\end{pmatrix}
$$

אלה הנתונים הקודמים עם שינוי אחד: כעת $T\begin{pmatrix}4\\-1\\2\end{pmatrix}=\begin{pmatrix}6\\3\\6\end{pmatrix}$ 
בהתאם לחישוב שנובע מהתלות הלינארית בין וקטורי הקלט. המשמעות היא שנתון זה מיותר, כי הוא נובע משני הנתונים האחרים. כדי למצוא נוסחה ל-
$T$
צריך להשלים את
$\Set{\begin{pmatrix}1\\-1\\1\end{pmatrix},\begin{pmatrix}2\\1\\0\end{pmatrix}}$
לבסיס ל-
$\R^3$.
דירוג לפי שורות מראה כי ניתן לעשות זאת ע"י הוספת הוקטור
$e_3$
(כי בסוף הדירוג לא יהיה איבר מוביל בעמודה השלישית).
הערך 
$T(e_3)$
נתון לבחירתנו, ולצורך נוחות נבחר
$T(e_3)=\begin{pmatrix}0\\0\\0\end{pmatrix}$.
נסמן
$$
.B=\Set{\begin{pmatrix}1\\-1\\1\end{pmatrix},\begin{pmatrix}2\\1\\0\end{pmatrix},\begin{pmatrix}0\\0\\1\end{pmatrix}}
$$

יהי
$\begin{pmatrix}x\\y\\z\end{pmatrix}\in\R^3$.
נחשב את
$\left[\begin{pmatrix}x\\y\\z\end{pmatrix}\right]_B$,
ואז נשתמש בקוארדינטות כדי לחשב את
$T\begin{pmatrix}x\\y\\z\end{pmatrix}$.
נדרג את המטריצה הבאה:
$$
\left(\begin{array}{ccc|c}
1 & 2 & 0 & x \\
-1 & 1 & 0 & y \\
1 & 0 & 1 & z
\end{array}\right)
$$
צורתה המדורגת קנונית היא:
$$
\left(\begin{array}{ccc|c}
1 & 0 & 0 & \frac{x-2y}{3} \\
0 & 1 & 0 & \frac{x+y}{3} \\
0 & 0 & 1 & \frac{-x+2y+3z}{3}
\end{array}\right)
$$

לכן, מתקיים
$$
.\begin{pmatrix}x\\y\\z\end{pmatrix}=\frac{x-2y}{3}\begin{pmatrix}1\\-1\\0\end{pmatrix}+\frac{x+y}{3}\begin{pmatrix}2\\1\\0\end{pmatrix}+\frac{-x+2y+3z}{3}\begin{pmatrix}0\\0\\1\end{pmatrix}
$$
נפעיל את
$T$
על שני האגפים ונקבל:
$$
\begin{aligned}
T\begin{pmatrix}x\\y\\z\end{pmatrix}&=\frac{x-2y}{3}T\begin{pmatrix}1\\-1\\1\end{pmatrix}+\frac{x+y}{3}T\begin{pmatrix}2\\1\\0\end{pmatrix}+\frac{-x+2y+3z}{3}T\begin{pmatrix}0\\0\\1\end{pmatrix} \\
&=\frac{x-2y}{3}\begin{pmatrix}1\\2\\3\end{pmatrix}+\frac{x+y}{3}\begin{pmatrix}4\\-1\\0\end{pmatrix}+\frac{-x+2y+3z}{3}\begin{pmatrix}0\\0\\0\end{pmatrix}=\begin{pmatrix}\frac{5x+2y}{3}\\\frac{x-5y}{3}\\x-2y\end{pmatrix}
\end{aligned}
$$
לסיכום, מצאנו דוגמה (אחת מתוך אינסוף) לה"ל שמקיימת את הנתונים:
$$
T\begin{pmatrix}x\\y\\z\end{pmatrix}=\begin{pmatrix}\frac{5x+2y}{3}\\\frac{x-5y}{3}\\x-2y\end{pmatrix}
$$

\end{enumerate}
\end{example}

\begin{exercise}
נסמן
$$
,v_1=\begin{pmatrix}1\\1\\0\end{pmatrix},\quad v_2=\begin{pmatrix}0\\1\\0\end{pmatrix},\quad v_3=\begin{pmatrix}0\\1\\1\end{pmatrix},\quad v_4=\begin{pmatrix}1\\1\\1\end{pmatrix}
$$
ובנוסף
$$
.w_1=\begin{pmatrix}2\\1\end{pmatrix},\quad w_2=\begin{pmatrix}2\\2\end{pmatrix},\quad w_3=\begin{pmatrix}1\\2\end{pmatrix},\quad w_4=\begin{pmatrix}1\\1\end{pmatrix}
$$
מצאו נוסחה לה"ל 
$T:\R^3\to\R^2$
המקיימת
$T(v_k)=w_k$
לכל
$1\leq k\leq 4$.
\end{exercise}

\begin{solution}
הקבוצה 
$\Set{v_1,v_2,v_3,v_4}$
בהכרח ת"ל כי יש בה ארבעה וקטורים - יותר מהמימד של 
$\R^3$.
דירוג מראה כי מתקיים
$$
.\begin{pmatrix}1\\1\\0\end{pmatrix}-\begin{pmatrix}0\\1\\0\end{pmatrix}+\begin{pmatrix}0\\1\\1\end{pmatrix}- \begin{pmatrix}1\\1\\1\end{pmatrix}=\begin{pmatrix}0\\0\\0\end{pmatrix}
$$
נבדוק שאין סתירה בנתונים. נפעיל את 
$T$
על שני האגפים, נציב את הנתונים ונקבל
$$
.\begin{pmatrix}2\\1\end{pmatrix}-\begin{pmatrix}2\\2\end{pmatrix}+\begin{pmatrix}1\\2\end{pmatrix}-\begin{pmatrix}1\\1\end{pmatrix}=\begin{pmatrix}0\\0\end{pmatrix}
$$

אם כן, אין סתירה וקיימת ויחידה
$T$
כנ"ל לפי משפט ההגדרה. ניתן לדלל כל אחד מהוקטורים
$v_1,v_2,v_3,v_4$
ולקבל בסיס ל-
$\R^3$.
נבחר את
$B=\Set{v_1,v_2,v_3}$.
כעת, יהי
$\begin{pmatrix}x\\y\\z\end{pmatrix}\in\R^3$.
נחשב וקטור קוארדינטות ע"י דירוג המטריצה הבאה:
$$
\left(\begin{array}{ccc|c}
1 & 0 & 0 & x \\
1 & 1 & 1 & y \\
0 & 0 & 1 & z
\end{array}\right)\xrightarrow{R_2\to R_2-R_1-R_3}\left(\begin{array}{ccc|c}
1 & 0 & 0 & x \\
0 & 1 & 0 & y-x-z \\
0 & 0 & 1 & z
\end{array}\right)
$$
לכן, מתקיים 
$\left[\begin{pmatrix}x\\y\\z\end{pmatrix}\right]_B=\begin{pmatrix}x\\y-x-z\\z\end{pmatrix}$
ובאופן שקול
$$
.\begin{pmatrix}x\\y\\z\end{pmatrix}=x\begin{pmatrix}1\\1\\0\end{pmatrix}+(y-x-z)\begin{pmatrix}0\\1\\0\end{pmatrix}+z\begin{pmatrix}0\\1\\1\end{pmatrix}
$$
נפעיל את 
$T$
על שני האגפים, נציב את הנתונים ונקבל
$$
.T\begin{pmatrix}x\\y\\z\end{pmatrix}=x\begin{pmatrix}2\\1\end{pmatrix}+(y-x-z)\begin{pmatrix}2\\2\end{pmatrix}+z\begin{pmatrix}1\\2\end{pmatrix}=\begin{pmatrix}2y-z\\2y-x\end{pmatrix}
$$
\end{solution}

\section{גרעין ותמונה}
אנחנו רגילים לפתור ממ"ליות. עבור
$A\in\M_{m\times n})(\F)$
הסתכלנו על ממ"ליות הומוגניות מהצורה
$Ax=\underline{0}$,
וראינו כי קבוצת הפתרונות היא למעשה תת-מרחב שקראנו לו מרחב הפתרונות
$\Null(A)\subseteq \F^n$.
לעומת זאת, גם ראינו כי קבוצת כל הוקטורים
$b\in\F^m$
עבורם יש פתרון לממ"ל
$Ax=b$,
היא מרחב העמודות 
$\Col(A)\subseteq \F^m$.

למעשה, מדובר במקרים פרטיים של שתי הגדרות יותר כלליות הקשורות להעתקה לינארית.

\begin{definition}
תהי
$T:V\to W$
ה"ל.

\begin{itemize}
\item 
הגרעין של 
$T$
מוגדר להיות
$$
.\Ker(T)=\Set{v\in V|T(v)=0_W}\subseteq V
$$

\item 
התמונה של 
$T$
מוגדרת להיות
$$
.\Image(T)=\Set{T(v)|v\in V}\subseteq W
$$


\end{itemize}
\end{definition}

% תמונה או הדגמה 

תמונה של פונקציה היא מושג כללי שנתקלים בו גם בקורסים מתמטיים אחרים (למשל חדו"א), ואפשר לחשוב עליה כעל קבוצת כל האיברים בטווח שיש להם מקורות בתחום (לפחות אחד לכל אחד). לעומת זאת, גרעין הוא קבוצת כל האיברים בתחום שנשלחים לוקטור האפס. בקורס שלנו גם לתמונה וגם לגרעין יש "אופי לינארי", כלומר שניהם הם תת-מרחבים.

\begin{proposition}
תהי
$T:V\to W$
ה"ל. אז מתקיים:
\begin{itemize}
    \item 
    $Ker(T)$
    היא תת-מרחב של 
    $V$.


    \item 
    $Im(T)$
    היא תת-מרחב של 
    $W$.

    
\end{itemize}
\end{proposition}

\begin{proof}
לכל קבוצה נבדוק סגירות לחיבור וכפל בסקלר.
\begin{enumerate}[label=\alph*.]
\item 
יהיו 
$v_1,v_2\in \Ker(T), \alpha\in\F$. 
מתקיים
$$
,T(v_1+v_2)=T(v_1)+T(v_2)=0_W+0_W=0_W
$$
ולכן
$v_1+v_2\in\Ker(T)$.
באופן דומה:
$$
,T(\alpha v_1)=\alpha T(v_1)=\alpha\cdot 0_W=0_W
$$
ולכן
$\alpha v_1\in\Ker(T)$.


\item
יהיו 
$w_1,w_2\in \Image(T), \alpha\in\F$. 
קיימים
$u_1,u_2\in V$
כך ש-
$$
.w_1=T(u_1),\quad w_2=T(u_2)
$$
נשים לב כי
$$
,T(u_1+u_2)=T(u_1)+T(u_2)=w_1+w_2
$$
ובפרט מתקיים
$w_1+w_2\in\Image(T)$.
באופן דומה:
$$
,T(\alpha u_1)=\alpha T(u_1)=\alpha w_1
$$
ולכן
$\alpha w_1\in\Image(T)$.


\end{enumerate}
\end{proof}

\begin{example}
תהי
$T:\M_{2\times 2}(\R)\to \R^2$
הנתונה ע"י
$$
.T\begin{pmatrix}
a & b \\
c & d
\end{pmatrix}=\begin{pmatrix}
a+b \\
c+d
\end{pmatrix}
$$
לפי הגדרת הגרעין, מתקיים:
$$
\begin{aligned}
\Ker(T)&=\Set{\begin{pmatrix}
a & b \\
c & d
\end{pmatrix}|\begin{pmatrix}
a+b \\
c+d
\end{pmatrix}=\begin{pmatrix}
0 \\
0
\end{pmatrix}}=\Set{\begin{pmatrix}
-s & s \\
-t & t
\end{pmatrix}|s,t\in \R} \\
&=\Span\left(\begin{pmatrix}
-1 & 1 \\
0 & 0
\end{pmatrix},\begin{pmatrix}
0 & 0 \\
-1 & 1
\end{pmatrix}\right)
\end{aligned}
$$
לפי הגדרת התמונה, מתקיים:
$$
\begin{aligned}
\Image(T)&=\Set{T\begin{pmatrix}
a & b \\
c & d
\end{pmatrix}|\begin{pmatrix}
a & b \\
c & d
\end{pmatrix}\in\M_{2\times 2}(\R)} \\
&=\Set{\begin{pmatrix}
a+b \\
c+d
\end{pmatrix}|a,b,c,d\in\R}=\Span\left(\begin{pmatrix}1\\0\end{pmatrix},\begin{pmatrix}0\\1\end{pmatrix}\right)=\R^2
\end{aligned}
$$
\end{example}

\begin{proposition}\label{prop:image-span}
תהי
$T:V\to W$
ה"ל. אם
$V=\Span(v_1,...,v_k)$,
אז מתקיים 
$$.\Image(T)=\Span(T(v_1),...,T(v_k))$$
\end{proposition}

\begin{proof}
לפי הנתון
$V=\Span(v_1,...,v_k)$,
לכל
$v\in V$
קיימים סקלרים
$\alpha_1,...,\alpha_k\in \F$
כך ש-
$$
.v=\alpha_1v_1+...+\alpha_kv_k
$$
לכן, לפי טענה
\ref{prop:more-linearity}
נובע כי:

$$
\begin{aligned}
\Image(T)&=\Set{T(v)|v\in V}=\Set{T(\alpha_1v_1+...+\alpha_kv_k)|\alpha_1,...,\alpha_k\in \F}\\
&=\Set{\alpha_1T(v_1)+...+\alpha_kT(v_k)|\alpha_1,...,\alpha_k\in \F}=\Span(T(v_1),...,T(v_k))
\end{aligned}
$$
\end{proof}

כמובטח, יש קשר בין גרעין ותמונה למרחב הפתרונות ומרחב העמודות של מטריצה.

\begin{proposition}\label{prop:matrix-map}
תהי 
$A\in\M_{m\times n}(\F)$.
אז מתקיים:
\begin{enumerate}[label=\alph*.]
\item
$$\Ker(T_A)=\Null(A)\subseteq \F^n$$

\item
$$\Image(T_A)=\Col(A)\subseteq \F^m$$


\end{enumerate}
\end{proposition}

\begin{proof}
נזכור כי
$T_A$
היא ה"ל מ-
$\F^n$
ל-
$\F^m$.
\begin{enumerate}[label=\alph*.]
\item
לפי הגדרת 
$T_A$,
לכל 
$v\in V$
מתקיים
$T_A(v)=Av$.
לכן:
$$
v\in\Ker(T)\iff T_A(v)=0 \iff Av=0 \iff v\in \Null(A)
$$
מכאן נובע כי 
$\Ker(T_A)=\Null(A)$.


\item 
לפי טענה 
\ref{prop:image-span},
מהעובדה 
$\F^n=\Span(e_1,...,e_n)$
נובע כי
$$
.\Image(T_A)=\Span(Ae_1,...,Ae_n)=\Col(A)
$$

השוויון האחרון נובע מכך שלכל 
$1\leq j\leq n$,
הוקטור 
$Ae_j$
הוא בדיוק העמודה ה-
$j$
של
$A$.


\end{enumerate}
\end{proof}

\begin{exercise}
תהי
$T:\R^4\to\R^3$
המוגדרת ע"י
$$
.T\begin{pmatrix}x\\y\\z\\w\end{pmatrix}=\begin{pmatrix}x+y+z\\2y+z+w\\x+y+3z\end{pmatrix}
$$
מצאו את
$\Ker(T),\Image(T)$.
\end{exercise}

\begin{solution}
מתקיים
$T=T_A$
כאשר
$$
.A=\begin{pmatrix}
1 & 1 & 1 & 0 \\
0 & 2 & 2 & 1 \\
1 & 1 & 3 & 0
\end{pmatrix}
$$
לפי טענה
\ref{prop:matrix-map}
מתקיים:
$$
\begin{aligned}
\Ker(T)&=\Ker(T_A)=\Null(A)\\
\Image(T)&=\Image(T_A)=\Col(A)
\end{aligned}
$$
נדרג קנונית ונקבל:
$$
\begin{pmatrix}
1 & 0 & 0 & -\tfrac12 \\
0 & 1 & 0 & \tfrac12 \\
0 & 0 & 1 & 0
\end{pmatrix}
$$
לכן, ע"י הצבת
$w=t$
נקבל
$$
.\Ker(T)=\Null(A)=\Set{\left(\frac{t}{2},-\frac{t}{2},0,t\right)|t\in\R}=\Span((1,-1,0,2))
$$
בנוסף, הדירוג מראה כי ניתן לקבל בסיס ל-
$\Col(A)$
ע"י דילול העמודה הרביעית של
$A$.
כך נקבל קבוצה של שלושה וקטורים בת"ל ב-
$\R^3$,
ולכן לפי משפט "שלישי חינם"
$$
.\Image(T)=\Col(A)=\R^3
$$
\end{solution}

לכל מ"ו יש העתקה פשוטה שניתן להגדיר באופן טבעי.

\begin{definition}
יהי
$V$
מ"ו מעל
$\F$.
נגדיר את העתקת הזהות
$\Id_V:V\to V$
ע"י
$$
,\Id_V(v)=v 
$$
כלומר ההעתקה ששולחת כל וקטור לעצמו.
\end{definition}

לכל שני מרחבים וקטוריים קיימת העתקה פשוטה ביניהם, שהגדרתה גם טבעית.

\begin{definition}
נגדיר את העתקת האפס
$\0_{V,W}:V\to V$
ע"י
$$
,\0_{V,W}(v)=0_W 
$$
כלומר ההעתקה הקבועה ששולחת כל וקטור ב-
$V$
לוקטור האפס ב-
$W$.
\end{definition}

\begin{remark}
עבור
$V=\F^n, W=\F^m$
מתקיים
$$
.\Id_V=T_{I_n},\quad \0_{V,W}=T_{\0_{m,n}}
$$
כלומר, העתקת הזהות מתאימה למטריצת הזהות, והעתקת האפס מתאימה למטריצת האפס.
\end{remark}

\begin{exercise}
יהיו
$V,W$
מרחבים וקטוריים מעל
$\F$.
\begin{enumerate}[label=\alph*.]
\item
הראו כי ההעתקות
$\0_{V,W},\Id_V$
הן לינאריות. 


\item
הראו כי מתקיים:
$$
\begin{aligned}
\Ker(\Id_V)      &= \Set{0_V},      & \quad \Image(\Id_V)      &= V \\
\Ker(\0_{V,W})   &= V,            & \quad \Image(\0_{V,W})   &= \Set{0_W}
\end{aligned}
$$
\end{enumerate}
\end{exercise}

\begin{solution}\
\begin{enumerate}[label=\alph*.]
\item 
יהיו 
$v_1,v_2\in V,\alpha\in\F$
מתקיים:
$$
\begin{aligned}
\Id_V(v_1+v_2)&=v_1+v_2=\Id(v_1)+\Id(v_2) \\
\0_{V,W}(v_1+v_2)&=0_W=0_W+0_W=\0_{V,W}(v_1)+\0_{V,W}(v_2)
\end{aligned}
$$

בנוסף:
$$
\begin{aligned}
\Id_V(\alpha v_1)&=\alpha v_1=\alpha\Id(v_1) \\
\0_{V,W}(\alpha v_1)&=0_W=\alpha\cdot 0_W=\alpha\0_{V,W}(v_1)
\end{aligned}
$$


\item
עבור העתקת הזהות:
$$
\begin{aligned}
\Ker(\Id_V)&=\Set{v\in V|\Id_V(v)=0_V}=\Set{v\in V|v=0_V}=\Set{0_V} \\
\Image(\Id_V)&=\Set{\Id_V(v)|v\in V}=\Set{v|v\in V}=V
\end{aligned}
$$

עבור העתקת האפס:
$$
\begin{aligned}
\Ker(\0_{V,W})&=\Set{v\in V|\0_{V,W}(v)=0_W}=\Set{v\in V|0_W=0_W}=V \\
\Image(\0_{V,W})&=\Set{\0_{V,W}(v)|v\in V}=\Set{0_W|v\in V}=\Set{0_W}
\end{aligned}
$$


\end{enumerate}
\end{solution}

המושגים הכלליים של "פונקציה על" ו"פונקציה חד-חד-ערכית" תקפים גם לגבי העתקות לינאריות. נגדיר אותם במדויק.

\begin{definition}
תהי
$T:V\to W$
ה"ל.
\begin{itemize}
\item
נאמר כי
$T$
על אם
$\Image(T)=W$.


\item 
נאמר כי
$T$
חד-חד-ערכית (חח"ע) אם לכל
$v_1,v_2\in V$
מתקיים
$$
.T(v_1)=T(v_2) \iff v_1=v_2
$$


\end{itemize}
\end{definition}

יש אפיון יותר פשוט להעתקה לינארית חח"ע:

\begin{proposition}\label{prop:injective}
תהי
$T:V\to W$ 
ה"ל. אז מתקיים
$$
.\Ker(T)=\Set{0_V} \iff \text{$T$ חח"ע}
$$

\end{proposition}

\begin{proof}
הכיוון 
$\impliedby$:
נניח כי
$T$
חח"ע.
נזכור כי
$T(0_V)=0_W$
לפי טענה
\ref{prop:more-linearity}.
לכן:
$$
,v\in\Ker(T) \iff T(v)=0_W \iff T(v)=T(0_V) \iff v=0_V
$$
ומכאן 
$\Ker(T)=\Set{0_V}$.

הכיוון 
$\implies$:
נניח כי
$\Ker(T)=\Set{0_V}$.
לכל
$v_1,v_2\in V$ 
מתקיים
$$
\begin{aligned}
T(v_1)=T(v_2) &\iff T(v_1)-T(v_2)=0_W \iff T(v_1-v_2)=0_W \\
&\iff v_1-v_2\in \Ker(T) \iff v_1-v_2=0_V \iff v_1=v_2
\end{aligned}
$$
קיבלנו כי
$T$
חח"ע.

\end{proof}

\begin{remark}
האפיון הזה לא חל על פונקציות לא לינאריות. למשל, הפונקציה
$f:\R\to\R$
המוגדרת ע"י
$f(x)=x^2$
אינה חח"ע כי
$f(1)=f(-1)$.
למרות זאת, קבוצת כל הפתרונות של המשוואה
$f(x)=0$
היא בדיוק
$\Set{0}$.

אם בקורס כמו חדו"א נדרשת חקירה (עם נגזרת) כדי להראות שפונקציה היא חח"ע, באלגברה לינארית הכלי הוא פתרון ממ"ל הומוגנית או שימוש בשיקולי מימד כדי להראות שיש פתרון טריוויאלי בלבד.
\end{remark}

\begin{proposition}\label{prop:matrix-map-2}
תהי
$A\in\M_{m\times n}(\F)$.
$T_A:\F^n\to\F^m$
מקיימת:
\begin{enumerate}[label=\alph*.]
\item 
$$
\begin{aligned}
\Null(A)=\Set{\underline{0}} &\iff \text{$T_A$ חח"ע} \\
\Col(A)=\F^m &\iff \text{$T_A$ על} 
\end{aligned}
$$


\item 
אם
$n=m$,
אז
$$
.\text{$T_A$ חח"ע} \iff \text{$T_A$ על} \iff \text{$A$ הפיכה}
$$


\end{enumerate}
\end{proposition}

\begin{proof}\
\begin{enumerate}[label=\alph*.]
\item
לפי טענות
\ref{prop:matrix-map}
ו-
\ref{prop:injective}
מתקיים:
$$
\begin{aligned}
\Null(A)=\Set{\underline{0}} &\iff \Ker(T_A)=\Set{\underline{0}} \iff \text{$T_A$ חח"ע} \\
\Col(A)=\F^m &\iff \Image(T_A)=\F^m \iff \text{$T_A$ על} 
\end{aligned}
$$


\item
עבור
$n=m$
ידוע כי (לפי משפט
\ref{thm:invertible2})
$$
,\Null(A)=\Set{\underline{0}} \iff \Col(A)=\F^m \iff \text{$A$ הפיכה}
$$
ולכן במקרה זה מתקיים
$$
.\text{$T_A$ חח"ע} \iff \text{$T_A$ על} \iff \text{$A$ הפיכה}
$$


\end{enumerate}
\end{proof}

\begin{example}
\begin{enumerate}[label=\alph*.]
נסתכל על
$T:\R^3\to \R^3$
המוגדרת ע"י
$$
.T\begin{pmatrix}x\\y\\z\end{pmatrix}=\begin{pmatrix}x+y\\y+z\\z+x\end{pmatrix}
$$
נשים לב כי
$T=T_A$
כאשר
$$
.A=\begin{pmatrix}
1 & 1 & 0 \\
0 & 1 & 1 \\
1 & 0 & 1
\end{pmatrix}
$$
זו מטריצה ריבועית, וניתן לבדוק כי
היא הפיכה (למשל כי
$\det(A)=2\neq 0$).
לכן, לפי טענה 
\ref{prop:matrix-map-2}
נובע כי
$T=T_A$
חח"ע ועל.

\end{enumerate}
\end{example}

המשפט הבא נקרא משפט המימדים להעתקות לינאריות.

\begin{theorem}\label{thm:ker-image}
תהי
$T:V\to W$
ה"ל. אז מתקיים
$$
.\dim(\Ker(T))+\dim(\Image(T))=\dim(V)
$$
\end{theorem}

\begin{remark}
משפט הדרגה (סעיף ב' של משפט
\ref{thm:rank-nullity})
הוא מקרה פרטי של משפט זה. נשים לב כי לכל 
$A\in\M_{m\times n}(\F)$
מתקיים
$$
,\Ker(T_A)=\Null(A),\quad \Image(T_A)=\Col(A) 
$$
ולכן
$$
.\dim(\Ker(T_A))=\dim(\Null(A)),\quad \dim(\Image(T_A))=\dim(\Col(A))=\rank(A) 
$$
בנוסף, התחום של 
$T_A$
הוא 
$\F^n$.
מכאן, לפי משפט המימדים עבור העתקות לינאריות נקבל
$$
.\dim(\Null(A))+\dim(\Col(A))=\dim(\F^n)=n
$$
\end{remark}

\begin{proof}
נסמן
$n=\dim(V),k=\dim(\Ker(T))$.

ל-
$\Ker(T)$
קיים בסיס
$B_0=\Set{v_1,...,v_k}$.
נשלים אותו לבסיס ל-
$V$ 
ע"י הוספת
$n-k$
וקטורים, ונקבל:
$$B_V=\Set{v_1,...,v_k,v_{k+1},...,v_n}$$
נשים לב כי
$$
,\Image(T)=\Span(T(v_1),...,T(v_n))=\Span(T(v_{k+1}),...,T(v_n))
$$
כאשר השוויון האחרון נובע מכך ש-
$T(v_j)=0_W$
לכל 
$1\leq j\leq k$
לפי ההנחה על
$B_0$.
אם נראה כי
$\Set{T(v_{k+1}),...,T(v_n)}$
היא קבוצה בת"ל, נקבל שהיא בסיס ל-
$\Image(T)$
ומכאן:
$$
\dim(\Image(T))=n-k=\dim(V)-\dim(\Ker(T))
$$
ובאופן שקול:
$$
\dim(\Ker(T))+\dim(\Image(T))=\dim(V)
$$

ובכן, נניח שקיימים 
$\alpha_{k+1},...,\alpha_n\in\F$
כך ש-
$$
.\alpha_{k+1}T(v_{k+1})+...+\alpha_nT(v_n)=0_W
$$
מלינאריות נובע כי
$$
,T(\alpha_{k+1}v_{k+1}+...+\alpha_nv_n)=0_W
$$
ולכן
$$
.\alpha_{k+1}v_{k+1}+...+\alpha_nv_n\in\Ker(T)
$$
נסמן 
$v=\alpha_{k+1}v_{k+1}+...+\alpha_nv_n$.
כעת,
$B_0=\Set{v_1,...,v_n}$
פורשת את
$\Ker(T)$.
בפרט, קיימים 
$\alpha_1,...,\alpha_k\in\F$
כך ש-
$v=\alpha_1v_1+...+\alpha_kv_k$.
נציב את הביטוי המקורי ונקבל
$$
.\alpha_{k+1}v_{k+1}+...+\alpha_nv_n=\alpha_1v_1+...+\alpha_kv_k
$$
ובאופן שקול:
$$
-\alpha_1v_1+...-\alpha_kv_k+\alpha_{k+1}v_{k+1}+...+\alpha_nv_n=0_W
$$
$B=\Set{v_1,...,v_n}$
בת"ל, ולכן בהכרח כל הסקלרים לעיל מתאפסים. בפרט, קיבלנו
$$\alpha_{k+1}=...=\alpha_n=0$$
כנדרש. אם כן, 
$\Set{T(v_{k+1}),...,T(v_n)}$
בת"ל וסיימנו את הוכחת המשפט.
\end{proof}

\begin{remark}
מהמשפט נובע כי
$$
.\dim(\Image(T))=\dim(V)-\dim(\Ker(T))\leq\dim(V)
$$
כבר היה ברור שמתקיים
$\dim(\Ker(T))\leq\dim(V)$
וגם
$\dim(\Image(T))\leq\dim(W)$
כי
$\Ker(T)\subseteq V,\Image(T)\subseteq W$.

יש גם דרך אינטואיטיבית לחשוב על אי-השוויון
$\dim(\Image(T))\leq\dim(V)$:
מימד התמונה לא יכול לעלות על מימד התחום כי 
$T$
שולחת כל וקטור בתחום לוקטור יחיד בתמונה. למשל, ישר לא יכול להפוך למישור באופן כזה.
\end{remark}

\begin{exercise}
הוכיחו או הפריכו:
\begin{enumerate}[label=\alph*.]
\item 
קיימת ה"ל
$T:\R^3\to\R^4$
שהיא על.

\item
קיימת ה"ל
$T:\R^4\to\R^3$
שהיא חח"ע.

\item
$T:\M_{2\times 3}(\C)\to\C^6$
היא על אם ורק אם היא חח"ע.


\end{enumerate}
\end{exercise}

\begin{solution}\
\begin{enumerate}[label=\alph*.]
\item 
נניח בשלילה שקיימת
$T:\R^3\to\R^4$
שהיא על. כלומר מתקיים
$\Image(T)=\R^4$,
ולפי משפט המימדים לה"ל נובע כי
$$
.\dim(\Ker(T))+\dim(\R^4)=\dim(\R^3)
$$
מכאן
$\dim(\Ker(T))=-1$
וזו סתירה. לסיכום, הראינו כי לא קיימת 
$T$
כזו.


\item
נניח שקיימת
$T:\R^4\to\R^3$
שהיא חח"ע. כלומר מתקיים
$\Ker(T)=\Set{\underline{0}}$,
ולפי משפט המימדים לה"ל נובע כי
$$
.0+\dim(\Image(T))=\dim(\R^4)
$$
קיבלנו
$\dim(\Image(T))=4$,
אבל מתקיים
$\Image(T)\subseteq\R^3$
ולכן
$$.\dim(\Image(T))\leq\dim(\R^3)=3$$
סתירה.


\item 
הטענה נכונה. נוכיח זאת: תהי
$T:\M_{2\times 3}(\C)\to\C^6$.
נשים לב כי
$$
.\dim(\M_{2\times 3}(\C))=6=\dim(\C^6)
$$
בנוסף, ממשפט המימדים לה"ל נובע כי
$$
,\dim(\Ker(T))+\dim(\Image(T))=6
$$
ולכן
$$
.\dim(\Ker(T))=0 \iff \dim(\Image(T))=6
$$
ובאופן שקול (לפי מסקנה
\ref{cor:subspace-dim}):
$$
.\Ker(T))=\Set{\underline{0}} \iff \Image(T)=\C^6
$$
לבסוף, לפי טענה
\ref{prop:injective}
אפשר להסיק כי 
$T$
היא על אם ורק אם היא חח"ע.

\end{enumerate}
\end{solution}

הטענה הבאה מכלילה את התרגיל האחרון:

\begin{proposition}
תהי
$T:V\to W$
ה"ל. 
\begin{enumerate}[label=\alph*.]
\item
אם
$T$
על,
אז
$$
.\dim(W)\leq\dim(V)
$$


\item
אם
$T$
חח"ע,
אז
$$
.\dim(W)\geq\dim(V)
$$


\end{enumerate}
\end{proposition}

\begin{proof}\
\begin{enumerate}[label=\alph*.]
\item
לפי הנתון, מתקיים 
$\Image(T)=W$.
לכן, לפי משפט המימדים לה"ל נובע כי
$$
.\dim(\Ker(T))+\dim(W)=\dim(V)
$$
מתקיים
$\dim(\Ker(T))\geq 0$,
ולכן:
$$
\dim(W)\leq\dim(V)
$$


\item 
הפעם מתקיים 
$\Ker(T)=\Set{0_V}$.
לפי המשפט המימדים נקבל
$$
.\dim(\Image(T))=\dim(V)
$$
מצד שני, 
$\Image(T)\subseteq W$
ולכן מתקיים
$\dim(\Image(T))\leq \dim(W)$.
בסך הכל:
$$
\dim(V)\leq\dim(W)
$$

\end{enumerate}
\end{proof}

אנחנו מוכנים להוכיח עוד משפט שראוי לשם "שלישי חינם". הפעם ההקשר הוא העתקות לינאריות.

\begin{theorem}\label{thm:3-for-2-iso}
תהי
$T:V\to W$
ה"ל. כל שניים מהתנאים הבאים גורר את התנאי השלישי:
\begin{enumerate}[label=\alph*.]
\item 
$T$
חח"ע.


\item
$T$
על.


\item 
$\dim(V)=\dim(W)$


\end{enumerate}
\end{theorem}

\begin{proof}
$(\text{ג})\impliedby(\text{ב})+(\text{א})$:
לפי ההנחות מתקיים
$$
.\dim(\Ker(T))=0,\quad \dim(\Image(T))=\dim(W)
$$
לאחר הצבה בנוסחה של משפט המימדים לה"ל, נקבל
$$
.\dim(W)=\dim(V)
$$

$(\text{ב})\impliedby(\text{ג})+(\text{א})$:
לפי ההנחות מתקיים
$$
.\dim(\Ker(T))=0,\quad \dim(V)=\dim(W)
$$
לאחר הצבה בנוסחה של משפט המימדים לה"ל, נקבל
$$
.\dim(\Image(T))=\dim(W)
$$
לפי מסקנה
\ref{cor:subspace-dim},
נובע כי
$\Image(T)=W$.
כלומר 
$T$
היא על.

$(\text{א})\impliedby(\text{ג})+(\text{ב})$:
לפי ההנחות מתקיים
$$
.\dim(\Image(T))=\dim(W),\quad \dim(V)=\dim(W)
$$
לאחר הצבה בנוסחה, נקבל
$$
.\dim(\Ker(T))+\dim(W)=\dim(W)
$$
נובע כי
$\dim(\Ker(T))=0$,
ולכן
$\Ker(T)=\Set{0_V}$.
פירוש הדבר ש-
$T$
חח"ע לפי טענה
\ref{prop:injective}.

\end{proof}

\begin{remark}
מה הקשר של המשפט לעיל לבין משפט "שלישי חינם" לקבוצת וקטורים (משפט 
\ref{thm:3-for-2})? 
המשפט הישן הוא מקרה פרטי של המשפט החדש. נראה זאת: תהי 
$A\in\M_{m\times n}(\F)$.
נסמן את עמודותיה ב-
$v_1,...,v_n$,
ונשים לב כי
$$
.\text{$v_1,...,v_n$ בת"ל}\iff \Null(A)=\Set{\underline{0}} \iff \Ker(T_A)=\Set{\underline{0}} \iff \text{$T$ חח"ע}
$$
בנוסף:
$$
.\text{$v_1,...,v_n$ פורשים את $\F^m$}\iff \Col(A)=\F^m \iff \Image(T_A)=\F^m \iff \text{$T$ על}
$$

לאור זאת ולפי המשפט החדש, כל שניים מהתנאים הבאים גוררים את התנאי השלישי:
\begin{enumerate}[label=\alph*.]
\item 
$v_1,...,v_n$
בת"ל.


\item
$\Span(v_1,...,v_n)=\F^m$


\item 
$n=m$


\end{enumerate}
זהו בדיוק משפט "שלישי חינם" המקורי לקבוצת וקטורים.

\end{remark}

\section{פעולות על העתקות לינאריות}

הפעולות הבסיסיות באלגברה לינארית הן חיבור וכפל בסקלר. נראה שניתן להגדיר פעולות דומות על העתקות לינאריות.

\begin{definition}
יהיו
$T,S:V\to W$
ה"ל ו-
$\alpha\in\F$.
\begin{itemize}
\item 
ההעתקה
$T+S:V\to W$
מוגדרת ע"י
$$
.(T+S)(v)=T(v)+S(v)
$$


\item 
ההעתקה
$\alpha\cdot T:V\to W$
מוגדרת ע"י
$$
.(\alpha\cdot T)(v)=\alpha\cdot T(v)
$$


\end{itemize}
\end{definition}

\begin{example}
יהיו 
$T,S:\M_{2\times 2}(\R)\to \M_{2\times 2}(\R)$
שמוגדרות ע"י
$$
.T(A)=A^t,\quad S\begin{pmatrix}a & b\\c & d\end{pmatrix}=\begin{pmatrix}c & d\\a & b\end{pmatrix}
$$
מתקיים:
$$
\begin{aligned}
(T+S)\begin{pmatrix}a & b\\c & d\end{pmatrix}&=T\begin{pmatrix}a & b\\c & d\end{pmatrix}+S\begin{pmatrix}a & b\\c & d\end{pmatrix}=\begin{pmatrix}a & c\\b & d\end{pmatrix}+\begin{pmatrix}c & d\\a & b\end{pmatrix}\\
&=\begin{pmatrix}a+c & c+d\\a+b & b+d\end{pmatrix}
\end{aligned}
$$
לכל
$\alpha\in\R$
מתקיים
$$
.(\alpha\cdot T)(A)=(\alpha A)^t=\alpha A^t
$$
ובאופן מפורש יותר:
$$
(\alpha\cdot T)\begin{pmatrix}a & b\\c & d\end{pmatrix}=\alpha\begin{pmatrix}a &c\\b & d\end{pmatrix}=\begin{pmatrix}\alpha a & \alpha c\\\alpha b & \alpha d\end{pmatrix}
$$
באופן דומה:
$$
(\alpha\cdot S)\begin{pmatrix}a & b\\c & d\end{pmatrix}=\alpha\begin{pmatrix}c & d\\a & b\end{pmatrix}=\begin{pmatrix}\alpha c & \alpha d\\\alpha a & \alpha b\end{pmatrix}
$$

\end{example}

\begin{exercise}
יהיו
$T,S:V\to W$
ה"ל ויהי
$\alpha\in\F$.
הוכיחו כי
$T+S,\alpha\cdot T$
הן העתקות לינאריות.
\end{exercise}

\begin{solution}
יהיו 
$v_1,v_2\in V, \beta\in \F$.
נבדוק תחילה את תכונת החיבור של הלינאריות עבור 
$T+S$.
נשתמש בהגדרה, בלינאריות של 
$T,S$
ולבסוף בחוק החילוף:
$$
\begin{aligned}
(T+S)(v_1+v_2)&=T(v_1+v_2)+S(v_1+v_2)=T(v_1)+T(v_2)+S(v_1)+S(v_2) \\
&=T(v_1)+S(v_1)+T(v_2)+S(v_2)=(T+S)(v_1)+(T+S)(v_2)
\end{aligned}
$$
נעבור לכפל בסקלר:
$$
\begin{aligned}
(T+S)(\beta\cdot v_1)&=T(\beta\cdot v_1)+S(\beta\cdot v_1)=\beta\cdot T(v_1)+\beta\cdot S(v_1)\\
&=\beta(T(v_1)+S(v_1))=\beta (T+S)(v_1)
\end{aligned}
$$

שתי הבדיקות עבור 
$\alpha\cdot T$
דומות. עבור חיבור נקבל:
$$
\begin{aligned}
(\alpha\cdot T)(v_1+v_2)&=\alpha\cdot T(v_1+v_2)=\alpha(T(v_1)+T(v_2)) \\
&=\alpha\cdot T(v_1)+\alpha\cdot T(v_2)=(\alpha\cdot T)(v_1)+(\alpha\cdot T)(v_2)
\end{aligned}
$$
ועבור כפל בסקלר:
$$
\begin{aligned}
(\alpha\cdot T)(\beta v_1)&=\alpha\cdot T(\beta v_1)=\alpha(\beta\cdot T(v_1)) \\
&=(\alpha\beta)T(v_1)=\beta(\alpha\cdot T(v_1))=\beta(\alpha\cdot T)(v_1)
\end{aligned}
$$

\end{solution}

\begin{remark}
מהו הקשר לפעולות החיבור והכפל בסקלר על מטריצות? נסתכל על המקרה הפרטי של
$T_A,T_B:\F^n\to\F^m$
המוגדרות ע"י מטריצות
$A,B\in\M_{m\times n}(\F)$.
לכל
$v\in\F^n$
מתקיים
$$
.(T_A+T_B)(v)=T_A(v)+T_B(v)=Av+Bv=(A+B)v=T_{A+B}(v)
$$
או באופן תמציתי:
$$
T_A+T_B=T_{A+B}
$$
באופן דומה, לכל
$v\in\F^n$
ולכלל
$\alpha\in \F$
מתקיים
$$
.(\alpha\cdot T_A)(v)=\alpha\cdot T_A(v)=\alpha\cdot Av=(\alpha\cdot A)v=T_{\alpha\cdot A}(v)
$$
ובקצרה:
$$
\alpha\cdot T_A=T_{\alpha\cdot A}
$$
לסיכום, הראינו שבמקרה הפרטי לעיל יש התאמה בין הפעולות על העתקות לינאריות לפעולות המתאימות על מטריצות. בקרוב נראה איך ניתן לייצג העתקה לינארית כללית ע"י מטריצה, ונוכיח את הקשר הכללי בין העתקות לינאריות, מטריצות והפעולות עליהן.
\end{remark}
בינתיים, נגדיר עוד פעולה שמוכרת מעולם הפונקציות:

\begin{definition}
יהיו
$T:V\to W$
ו-
$S:W\to U$
העתקות לינאריות. אז ההרכבה
$S\circ T:V\to U$
מוגדרת ע"י
$$
.S\circ T(v)=S(T(v))
$$
\end{definition}

% הדגמה או ציור 

\begin{example}\label{example:composition}\
\begin{enumerate}[label=\alph*.]
\item
יהיו
$T:\R^2\to\R^2$
ו-
$S:\R^2\to \R^3$
המוגדרות ע"י
$$
.S\begin{pmatrix}x\\y\end{pmatrix}=\begin{pmatrix}2x\\3y\\4x-y\end{pmatrix},\quad T\begin{pmatrix}x\\y\end{pmatrix}=\begin{pmatrix}x+y\\x-y\end{pmatrix}
$$
ההרכבה
$S\circ T:\R^2\to \R^3$
מוגדרת ומתקיים
$$
\begin{aligned}
S\circ T\begin{pmatrix}x\\y\end{pmatrix}&=S\left(T\begin{pmatrix}x\\y\end{pmatrix}\right)=S\begin{pmatrix}x+y\\x-y\end{pmatrix}=\begin{pmatrix}2(x+y)\\3(x-y)\\4(x+y)-(x-y)\end{pmatrix} \\
&=\begin{pmatrix}2x+2y\\3x-3y\\3x+5y\end{pmatrix}
\end{aligned}
$$

\item
יהיו
$T:\R^3\to\M_{2\times 2}(\R)$
ו-
$S:\M_{2\times 2}(\R)\to \R^4$
המוגדרות ע"י
$$
.S\begin{pmatrix}
a & b \\
c & d
\end{pmatrix}=\begin{pmatrix}a\\a+b\\b+c\\c+d\end{pmatrix},\quad T\begin{pmatrix}x\\y\\z\end{pmatrix}=\begin{pmatrix}
x & x+z \\
y+z & y
\end{pmatrix}
$$
ההרכבה 
$S\circ T:\R^3\to\R^4$
מוגדרת ומתקיים
$$
\begin{aligned}
S\circ T\begin{pmatrix}x\\y\\z\end{pmatrix}&=S\left(T\begin{pmatrix}x\\y\\z\end{pmatrix}\right)=S\begin{pmatrix}
x & x+z \\
y+z & y
\end{pmatrix}=\begin{pmatrix}x\\x+(x+z)\\(x+z)+(y+z)\\(y+z)+y\end{pmatrix} \\
&=\begin{pmatrix}x\\2x+z\\x+y+2z\\2y+z\end{pmatrix}
\end{aligned}
$$


\end{enumerate}
\end{example}

\begin{exercise}
יהיו
$T:V\to W$
ו-
$S:W\to U$
ה"ל. הוכיחו כי
$S\circ T$
היא ה"ל.
\end{exercise}

\begin{solution}
יהיו
$v_1,v_2\in V$
ו-
$\alpha\in\F$.
נשתמש בלינאריות של 
$T,S$
ונקבל:
$$
\begin{aligned}
S\circ T(v_1+v_2)&=S(T(v_1+v_2))=S(T(v_1)+T(v_2))=S(T(v_1))+S(T(v_2)) \\
&=S\circ T(v_1)+S\circ T(v_2)
\end{aligned}
$$
באופן דומה:
$$
S\circ T(\alpha v_1)=S(T(\alpha v_1))=S(\alpha T(v_1))=\alpha S(T(v_1))=\alpha S\circ T(v_1)
$$

\end{solution}

כעת נראה שבמקרה הפרטי של העתקות לינאריות המוגדרות ע"י מטריצות, ההרכבה (אם היא מוגדרת) מתאימה לכפל מטריצות.

\begin{proposition}\label{prop:homeo}
יהיו
$A\in\M_{m\times n}(\F),B\in\M_{n\times k}(\F)$.
עבור
$T_A:\F^n\to\F^m,T_B:\F^k\to\F^n$
ההרכבה
$T_B\circ T_A:\F^k\to\F^m$
מוגדרת ומתקיים
$$
.T_A\circ T_B=T_{AB}
$$
\end{proposition}

\begin{proof}
לכל
$v\in\F^k$
מתקיים
$$
.T_A\circ T_B(v)=T_A(T_B(v))=T_A(Bv)=A(Bv)=(AB)v=T_{AB}(v)
$$
ולכן
$$
.T_A\circ T_B=T_{AB}
$$
\end{proof}

\begin{example}
נחזור לסעיף א' בדוגמה
\ref{example:composition}.
שם מתקיים
$S=T_A,T=T_B$
כאשר
$$
.A=\begin{pmatrix}
2 & 0 \\
0 & 3 \\
4 & -1
\end{pmatrix},\quad B=\begin{pmatrix}
1 & 1 \\
1 & -1
\end{pmatrix}
$$
מתקיים
$$
.AB=\begin{pmatrix}
2 & 2 \\
3 & -3 \\
3 & 5
\end{pmatrix}
$$
זו דרך נוספת לראות כי מתקיים
$$
.S\circ T\begin{pmatrix}x\\y\end{pmatrix}=T_A\circ T_B\begin{pmatrix}x\\y\end{pmatrix}=T_{AB}\begin{pmatrix}x\\y\end{pmatrix}=\begin{pmatrix}
2 & 2 \\
3 & -3 \\
3 & 5
\end{pmatrix}\begin{pmatrix}x\\y\end{pmatrix}=\begin{pmatrix}2x+2y\\3x-3y\\3x+5y\end{pmatrix}
$$

\end{example}

\begin{remark}
יש אנלוגיה בין הרכבה של העתקות לינאריות לבין כפל מטריצות. בפרט, אין חוק חילוף להרכבה גם אם לשתי ההרכבות יש אותו תחום וטווח. כלומר, עבור ה"ל 
$S,T:V\to V$
לא בהכרח מתקיים 
$T\circ S=S\circ T$.
בתור דוגמה נגדית ניקח
$V=\R^2$
עם
$$
.T\begin{pmatrix}x\\y\end{pmatrix}=\begin{pmatrix}y\\x\end{pmatrix},\quad S\begin{pmatrix}x\\y\end{pmatrix}=\begin{pmatrix}x\\0\end{pmatrix}
$$
נחשב את ההרכבות ונקבל:
$$
\begin{aligned}
T\circ S\begin{pmatrix}x\\y\end{pmatrix}&=T\begin{pmatrix}x\\0\end{pmatrix}=\begin{pmatrix}0\\x\end{pmatrix} \\
S\circ T\begin{pmatrix}x\\y\end{pmatrix}&=S\begin{pmatrix}y\\x\end{pmatrix}=\begin{pmatrix}y\\0\end{pmatrix}
\end{aligned}
$$
אם כן, מתקיים
$T\circ S\neq S\circ T$.
\end{remark}

התרגיל הבא מראה קשר בין הרכבה לגרעין ותמונה.

\begin{exercise}\label{exer:composition}
יהיו
$T:V\to W$
ו-
$S:W\to U$
ה"ל. הוכיחו כי:
\begin{enumerate}[label=\alph*.]
\item 
$$
\Ker(T)\subseteq \Ker(S\circ T)
$$


\item 
$$
\Image(S\circ T)\subseteq \Image(S)
$$


\item 
אם
$S\circ T$
חח"ע, אז
$T$
חח"ע.


\item 
אם
$S\circ T$,
אז 
$S$
על.


\end{enumerate}
\end{exercise}

\begin{solution}\
\begin{enumerate}[label=\alph*.]
\item 
יהי
$v\in\Ker(T)$.
לפי ההגדרות וטענה
\ref{prop:more-linearity},
מתקיים
$$
.S\circ T(v)=S(T(v))=S(0_W)=0_U
$$
כלומר
$v\in\Ker(S\circ T)$,
ולכן:
$$
\Ker(T)\subseteq \Ker(S\circ T)
$$


\item 
יהי
$w\in\Image(S\circ T)$.
לפי ההגדרה, קיים
$v\in V$
כך ש-
$$
.w=S\circ T(v)=S(T(v))
$$
זה מראה כי 
$S$
שולחת את
$T(v)$
ל-
$w$,
ולכן
$w\in\Image(S)$. 
מכאן:
$$
\Image(S\circ T)\subseteq \Image(S)
$$


\item 
נניח כי
$S\circ T$
חח"ע. לפי טענה
\ref{prop:injective}
מתקיים
$\Ker(S\circ T)=\Set{0_V}$,
ולפי סעיף א' נובע כי
$$
.\Set{0_V}\subseteq \Ker(T)\subseteq \Ker(S\circ T)=\Set{0_V}
$$
מכאן
$\Ker(T)=\Set{0_V}$,
ושוב לפי הטענה (בכיוון ההפוך שלה) נובע כי
$T$
חח"ע.


\item 
נניח כי
$S\circ T$
על, כלומר 
$\Image(S\circ T)=U$.
לפי סעיף ב' נובע כי
$$
.U=\Image(S\circ T)\subseteq \Image(S)=U
$$
לכן
$\Image(S)=U$,
כלומר 
$S$
על.


\end{enumerate}
\end{solution}

\section{מטריצות מייצגות של העתקות לינאריות}
יהי
$V$
מ"ו נוצר סופית (בעל בסיס) מעל 
$\F$.
בפרק
\ref{ch:vectorspaces}
דיברנו על וקטורי קוארדינטות, שהם דרך להציג וקטורים ב-
$V$
באופן מוכר יותר כוקטורים ב-
$\F^n$.
זו דרך נוחה לבדוק מושגים כמו פרישה ותלות לינארית (או אי-תלות לינארית) ע"י דירוג המטריצה המתאימה.
בהינתן בסיס סדור
$B=\Set{v_1,...,v_n}$
של
$V$,
לכל וקטור
$v\in V$
יש סקלרים יחידים
$\alpha_1,...,\alpha_n\in\F$
עבורם מתקיים
$$
.v=\alpha_1 v_1+...+\alpha_n v_n
$$
בהתאם, הגדרנו את וקטור הקוארדינטות של 
$v$
לפי 
$B$:
$$
[v]_B=\begin{pmatrix}\alpha_1\\\vdots\\\alpha_n\end{pmatrix}
$$

\begin{example}
ב-
$\R^3$
נסתכל על הבסיס הסדור
$$
.B=\Set{\begin{pmatrix}1\\1\\1\end{pmatrix},\begin{pmatrix}1\\-1\\1\end{pmatrix},\begin{pmatrix}1\\1\\-1\end{pmatrix}}
$$
מתקיים
$$
,2\begin{pmatrix}1\\1\\1\end{pmatrix}+1\begin{pmatrix}1\\-1\\1\end{pmatrix}+3\begin{pmatrix}1\\1\\-1\end{pmatrix}=\begin{pmatrix}6\\4\\0\end{pmatrix}
$$
ולכן
$$
.\left[\begin{pmatrix}6\\4\\0\end{pmatrix}\right]_B=\begin{pmatrix}2\\1\\3\end{pmatrix}
$$
\end{example}

כעת נרצה להשתמש ברעיון של וקטורי קוארדינטות כדי לייצג העתקות לינאריות ע"י מטריצות, מה שיאפשר לנו לעשות חישובים יותר בקלות (למשל, עבור גרעין ותמונה).

\begin{definition}
יהיו
$T:V\to W$,
בסיס סדור
$B=\Set{v_1,...,v_n}$
ל-
$V$
ובסיס סדור
$C$
ל-
$W$.
נגדיר את המטריצה המייצגת של
$T$
לפי
$B,C$
ע"י
$$
.[T]_C^B=\begin{pmatrix}
\begin{array}{c}
|\\[-0.4ex]
[T(v_1)]_C\\[-0.4ex]
|
\end{array}
&
\begin{array}{c}
|\\[-0.4ex]
[T(v_2)]_C\\[-0.4ex]
|
\end{array}
&
\cdots
&
\begin{array}{c}
|\\[-0.4ex]
[T(v_n)]_C\\[-0.4ex]
|
\end{array}
\end{pmatrix}
$$
\end{definition}

\begin{example}\
\begin{enumerate}[label=\alph*.]
\item
תהי
$T:\R^2\to \R^3$
המוגדרת ע"י
$$
.T\begin{pmatrix}x\\y\end{pmatrix}=\begin{pmatrix}x+2y\\2x+y\\x-y\end{pmatrix}
$$
ניקח את הבסיסים הסטנדרטיים
$$
E_2=\Set{\begin{pmatrix}1\\0\end{pmatrix},\begin{pmatrix}0\\1\end{pmatrix}},E_3=\Set{\begin{pmatrix}1\\0\\0\end{pmatrix},\begin{pmatrix}0\\1\\0\end{pmatrix},\begin{pmatrix}0\\0\\1\end{pmatrix}}

הוא בסיס סטנדרטי.

\item 
הכללה של הסעיף הקודם: תהי

כאשר

|
\end{array}
&

|\\[-0.4ex]

&

&=\begin{pmatrix}
\begin{array}{c}
|\\[-0.4ex]

|

|\\[-0.4ex]

היא לפי הבסיסים הסטנדרטיים. המטריצה המייצגת במקרה זה היא

עצמה.


\item 
תהי
$T:\M_{2\times 2}(\F)\to\M_{2\times 2}(\F)$
העתקת השחלוף המוגדרת ע"י

$$
ניקח את הבסיס הסטנדרטי

נשאלת השאלה: מהו השימוש של מטריצה מייצגת? במובן הבסיסי ביותר, זוהי דרך לחשוב על העתקה לינארית כעל כפל במטריצה, שזו פעולה שכבר התרגלנו אליה. הטענה הבאה תבהיר את הכוונה המדויקת.

\begin{proposition}\label{prop:rep-matrix}
יהיו
$T:V\to W$,
$S:W\to U$
ה"ל, ויהיו
$B,C,D$
בסיסים סדורים ל-
$V,W,U$
בהתאמה. אז:

\begin{enumerate}[label=\alph*.]
\item 
לכל 
$v\in V$
מתקיים
$$
.[T(v)]_C=[T]_C^B[v]_B
$$


\item 
$$
[S\circ T]_D^B=[S]_D^C[T]_C^B
$$


\end{enumerate}
\end{proposition}

\begin{proof}\
\begin{enumerate}[label=\alph*.]
\item
נסמן
$B=\Set{v_1,...,v_n}$.
לכל
$v\in V$
קיימים סקלרים יחידים
$\alpha_1,...,\alpha_n\in\F$
כך ש-
$$
,v=\alpha_1 v_1+...+\alpha_n v_n
$$
ובהתאם מתקיים
$$
.[v]_B=\begin{pmatrix}\alpha_1\\\vdots\\\alpha_n\end{pmatrix}
$$
לכן, מהלינאריות של
$T$
וגם של קוארדינטות נובע כי
$$
\begin{aligned}
[T(v)]_C&=[\alpha_1T(v_1)+...+\alpha_nT(v_n)]_C=\alpha_1[T(v_1)]_C+...+\alpha_n[T(v_n)]_C \\
&=\begin{pmatrix}
\begin{array}{c}
|\\[-0.4ex]
[T(v_1)]_C\\[-0.4ex]
|
\end{array}
&
\begin{array}{c}
|\\[-0.4ex]
[T(v_2)]_C\\[-0.4ex]
|
\end{array}
&
\cdots
&
\begin{array}{c}
|\\[-0.4ex]
[T(v_n)]_C\\[-0.4ex]
|
\end{array}
\end{pmatrix}
\begin{pmatrix}\alpha_1\\\vdots\\\alpha_n\end{pmatrix}=[T]_C^B[v]_B
\end{aligned}
$$
\item 
תחילה נבצע את כפל המטריצות בהתאם לעמודות של המטריצה הימנית במכפלה, ואחר כך נשתמש בסעיף הקודם:
$$
\begin{aligned}
[S]_D^C[T]_C^B&=[S]_D^C\begin{pmatrix}
\begin{array}{c}
|\\[-0.4ex]
[T(v_1)]_C\\[-0.4ex]
|
\end{array}
&
\begin{array}{c}
|\\[-0.4ex]
[T(v_2)]_C\\[-0.4ex]
|
\end{array}
&
\cdots
&
\begin{array}{c}
|\\[-0.4ex]
[T(v_n)]_C\\[-0.4ex]
|
\end{array}
\end{pmatrix}\\
&=\begin{pmatrix}
\begin{array}{c}
|\\[-0.4ex]
[S]_D^C[T(v_1)]_C\\[-0.4ex]
|
\end{array}
&
\begin{array}{c}
|\\[-0.4ex]
[S]_D^C[T(v_2)]_C\\[-0.4ex]
|
\end{array}
&
\cdots
&
\begin{array}{c}
|\\[-0.4ex]
[S]_D^C[T(v_n)]_C\\[-0.4ex]
|
\end{array}
\end{pmatrix} \\
&=\begin{pmatrix}
\begin{array}{c}
|\\[-0.4ex]
[S\circ T(v_1)]_D\\[-0.4ex]
|
\end{array}
&
\begin{array}{c}
|\\[-0.4ex]
[S\circ T(v_2)]_D\\[-0.4ex]
|
\end{array}
&
\cdots
&
\begin{array}{c}
|\\[-0.4ex]
[S\circ T(v_n)]_D\\[-0.4ex]
|
\end{array}
\end{pmatrix}\\
&=[S\circ T]_D^B
\end{aligned}
$$


\end{enumerate}
\end{proof}

\begin{example}
יהיו
$T,S:\R^2\to\R^2$
ה"ל המקיימות את הנתונים הבאים:
$$
\begin{alignedat}{2}
T\begin{pmatrix}1\\1\end{pmatrix}  &= \begin{pmatrix}2\\3\end{pmatrix},\quad &
T\begin{pmatrix}1\\-1\end{pmatrix} &= \begin{pmatrix}3\\-1\end{pmatrix} \\
S\begin{pmatrix}3\\1\end{pmatrix}  &= \begin{pmatrix}2\\2\end{pmatrix},\quad &
S\begin{pmatrix}1\\3\end{pmatrix} &= \begin{pmatrix}0\\4\end{pmatrix}
\end{alignedat}
$$
ניתן למצוא נוסחאות ל-
$S,T$
ולחשב את
$S\circ T$,
אך נפעל בדרך אחרת. נסתכל על שלושת הבסיסים הסדורים הבאים ל-
$\R^2$:
$$
B=\Set{\begin{pmatrix}1\\1\end{pmatrix},\begin{pmatrix}1\\-1\end{pmatrix}},\quad C=\Set{\begin{pmatrix}3\\1\end{pmatrix},\begin{pmatrix}1\\3\end{pmatrix}}, \quad D=\Set{\begin{pmatrix}2\\-1\end{pmatrix},\begin{pmatrix}1\\2\end{pmatrix}}
$$
בחרנו את
$B,C$
בהתאם לוקטורי הקלט של הנתונים, ואילו 
$D$
הוא בחירה שרירותית לצורך הדוגמה. כעת נחשב את המטריצות המייצגות
$[S]_D^C,[T]_C^B$.
תחילה:
$$
[T]_C^B=\begin{pmatrix}
\begin{array}{c}
|\\[-0.4ex]
\left[T\begin{pmatrix}1\\1\end{pmatrix}\right]_C\\[-0.4ex]
|
\end{array}
&
\begin{array}{c}
|\\[-0.4ex]
\left[T\begin{pmatrix}1\\-1\end{pmatrix}\right]_C\\[-0.4ex]
|
\end{array}
\end{pmatrix}=\begin{pmatrix}
\begin{array}{c}
|\\[-0.4ex]
\left[\begin{pmatrix}2\\3\end{pmatrix}\right]_C\\[-0.4ex]
|
\end{array}
&
\begin{array}{c}
|\\[-0.4ex]
\left[\begin{pmatrix}3\\-1\end{pmatrix}\right]_C\\[-0.4ex]
|
\end{array}
\end{pmatrix}
$$
נחשב את וקטורי הקוארדינטות ע"י דירוג המטריצה המורחבת:
$$
\begin{pmatrix}
3 & 1 & \vline & 2 & \vline & 3 \\
1 & 3 & \vline & 3 & \vline & -1
\end{pmatrix}
$$

לאחר דירוג קנוני נקבל את המטריצה הבאה:
$$
\begin{pmatrix}
1 & 0 & \vline & \frac{3}{8} & \vline & \frac{5}{4} \\
0 & 1 & \vline & \frac{7}{8} & \vline & -\frac{3}{4}
\end{pmatrix}
$$

ולכן:
$$
[T]_C^B=\begin{pmatrix}
\frac{3}{8} & \frac{5}{4} \\
\frac{7}{8} & -\frac{3}{4}
\end{pmatrix}
$$

נעבור למטריצה המייצגת הבאה:
$$
[S]_D^C=\begin{pmatrix}
\begin{array}{c}
|\\[-0.4ex]
\left[S\begin{pmatrix}3\\1\end{pmatrix}\right]_C\\[-0.4ex]
|
\end{array}
&
\begin{array}{c}
|\\[-0.4ex]
\left[S\begin{pmatrix}1\\3\end{pmatrix}\right]_D\\[-0.4ex]
|
\end{array}
\end{pmatrix}=\begin{pmatrix}
\begin{array}{c}
|\\[-0.4ex]
\left[\begin{pmatrix}2\\2\end{pmatrix}\right]_D\\[-0.4ex]
|
\end{array}
&
\begin{array}{c}
|\\[-0.4ex]
\left[\begin{pmatrix}0\\4\end{pmatrix}\right]_D\\[-0.4ex]
|
\end{array}
\end{pmatrix}
$$
שוב נחשב את וקטורי הקוארדינטות ע"י דירוג מטריצה מורחבת:
$$
\begin{pmatrix}
2 & 1 & \vline & 2 & \vline & 0 \\
-1 & 2 & \vline & 2 & \vline & 4
\end{pmatrix}
$$
\end{example}

צורתה המדורגת קנונית היא:
$$
\begin{pmatrix}
1 & 0 & \vline & \frac{2}{5} & \vline & -\frac{4}{5} \\
0 & 1 & \vline & \frac{6}{5} & \vline & \frac{8}{5}
\end{pmatrix}
$$
מכאן:
$$
[S]_D^C=\begin{pmatrix}
\frac{2}{5} & -\frac{4}{5} \\
\frac{6}{5} & \frac{8}{5}
\end{pmatrix}
$$
לבסוף, לפי טענה 
\ref{prop:rep-matrix}
נובע כי
$$
.[S\circ T]_D^B=[S]_D^C[T]_C^B=\begin{pmatrix}
\frac{2}{5} & -\frac{4}{5} \\
\frac{6}{5} & \frac{8}{5}
\end{pmatrix}\begin{pmatrix}
\frac{3}{8} & \frac{5}{4} \\

\end{pmatrix}=\begin{pmatrix}
-\frac{11}{20} & \frac{11}{10} \\
\frac{37}{20} & \frac{3}{10}
\end{pmatrix}
$$

יהיו
$T,S:V\to W$
ה"ל, ויהיו
$B,C$
בסיסים סדורים ל-
$V,W$
בהתאמה. אז:
\begin{enumerate}[label=\alph*.]

$$

$$

יהי

\subsection{מטריצת מעבר בין בסיסים}

החישוב של מטריצה מייצגת עלול להיות ארוך אם לפחות אחד מהבסיסים אינו סטנדרטי. הכלי הבא עשוי לקצר את החישובים, ונראה בהמשך שיש לו חשיבות גם מעבר לכך.

\begin{definition}
יהי
$V$
מ"ו נוצר סופית מעל
$\F$
עם בסיסים (סדורים)
$B,C$.
מטריצת המעבר בין הבסיסים
$B$
ו-
$C$
מוגדרת ע"י
$$
.[I]_C^B=[\Id_V]_C^B
$$
\end{definition}

\begin{example}\
\begin{enumerate}[label=\alph*.]
\item
עבור
$C=B=\Set{v_1,...,v_n}$
מתקיים:
$$
\begin{aligned}
[I]_B^B&=\begin{pmatrix}
\begin{array}{c}
|\\[-0.4ex]
[\Id_V(v_1)]_B\\[-0.4ex]
|
\end{array}
&
\begin{array}{c}
|\\[-0.4ex]
[\Id_V(v_2)]_B\\[-0.4ex]
|
\end{array}
&
\cdots
&
\begin{array}{c}
|\\[-0.4ex]
[\Id_V(v_n)]_B\\[-0.4ex]
|
\end{array}
\end{pmatrix}\\
&=\begin{pmatrix}
\begin{array}{c}
|\\[-0.4ex]
[v_1]_B\\[-0.4ex]
|
\end{array}
&
\begin{array}{c}
|\\[-0.4ex]
[v_2]_B\\[-0.4ex]
|
\end{array}
&
\cdots
&
\begin{array}{c}
|\\[-0.4ex]
[v_n]_B\\[-0.4ex]
|
\end{array}
\end{pmatrix}=I_{n\times n}
\end{aligned}
$$
כלומר מטריצת המעבר מבסיס (בגודל
$n$)
לעצמו היא מטריצת היחידה
$I_{n\times n}$.


\item 
יהי
$E$
הבסיס הסטנדרטי ל-
$\R^3$
וניקח בנוסף את הבסיס הסדור
$$
.B=\Set{\begin{pmatrix}1\\2\\3\end{pmatrix},\begin{pmatrix}1\\3\\7\end{pmatrix},\begin{pmatrix}0\\2\\1\end{pmatrix}}
$$
ייצוג וקטורים לפי 
$E$
הוא מיידי, ולכן נקבל
$$
\begin{aligned}
[I]_E^B&=\begin{pmatrix}
\begin{array}{c}
|\\[-0.4ex]
\left[\begin{pmatrix}1\\2\\3\end{pmatrix}\right]_E\\[-0.4ex]
|
\end{array}
&
\begin{array}{c}
|\\[-0.4ex]
\left[\begin{pmatrix}1\\3\\7\end{pmatrix}\right]_E\\[-0.4ex]
|
\end{array}
&
\begin{array}{c}
|\\[-0.4ex]
\left[\begin{pmatrix}0\\2\\1\end{pmatrix}\right]_E\\[-0.4ex]
|
\end{array}
\end{pmatrix} \\
&=\begin{pmatrix}
1 & 1 & 0 \\
2 & 3 & 2 \\
3 & 7 & 1 
\end{pmatrix}
\end{aligned}
$$

מה לגבי
$[I]_B^E$?
זהו חישוב יותר ארוך שדורש דירוג.

$$
[I]_B^E=\begin{pmatrix}
\begin{array}{c}
|\\[-0.4ex]
\left[\begin{pmatrix}1\\0\\0\end{pmatrix}\right]_B\\[-0.4ex]
|
\end{array}
&
\begin{array}{c}
|\\[-0.4ex]
\left[\begin{pmatrix}0\\1\\0\end{pmatrix}\right]_B\\[-0.4ex]
|
\end{array}
&
\begin{array}{c}
|\\[-0.4ex]
\left[\begin{pmatrix}0\\0\\1\end{pmatrix}\right]_B\\[-0.4ex]
|
\end{array}
\end{pmatrix}
$$
לצורך חישוב וקטורי הקוארדינטות, נדרג את המטריצה הבאה:
$$
\begin{pmatrix}
1 & 1 & 0 & \vline & 1 & \vline & 0 & \vline & 0 \\
2 & 3 & 2 & \vline & 0 & \vline & 1 & \vline & 0 \\
3 & 7 & 1 & \vline & 0 & \vline & 0 & \vline & 1
\end{pmatrix}
$$
נשים לב שזו בדיוק הדרך לחשב את המטריצה ההופכית של המטריצה שעמודותיה הן וקטורי
$B$
(זה לא מקרי).
לאחר דירוג קנוני נקבל:
$$
\begin{pmatrix}
1 & 0 & 0 & \vline & \frac{11}{7} & \vline & \frac{1}{7} & \vline & -\frac{2}{7} \\
0 & 1 & 0 & \vline & -\frac{4}{7} & \vline & -\frac{1}{7} & \vline & \frac{2}{7} \\
0 & 0 & 1 & \vline & -\frac{5}{7} & \vline & \frac{4}{7} & \vline & -\frac{1}{7}
\end{pmatrix}
$$
ולכן:
$$
[I]_B^E=\begin{pmatrix}
\frac{11}{7} & \frac{1}{7} & -\frac{2}{7} \\
-\frac{4}{7} & -\frac{1}{7} & \frac{2}{7} \\
-\frac{5}{7} & \frac{4}{7} & -\frac{1}{7}
\end{pmatrix}
$$

\end{enumerate}
\end{example}

הדוגמה האחרונה היא הקדמה לטענה הבאה:

\begin{proposition}\label{prop:rep-inverse}
יהי
$V$
מ"ו מעל
$\F$,
ויהיו
$B,C$
בסיסים סדורים ל-
$V$.
אז 
$[I]_C^B,[I]_B^C$
הפיכות ומתקיים
$$
.([I]_C^B)^{-1}=[I]_B^C
$$
\end{proposition}

\begin{proof}
נניח כי
$\dim(V)=n$.
לפי טענה
\ref{prop:rep-matrix}
מתקיים
$$
.[I]_B^C[I]_C^B=[\Id_V]_B^C[\Id_V]_C^B=[\Id_V\circ\Id_V]_B^B=[\Id_V]_B^B=I_{n\times n}
$$
לכן, שתי המטריצות הפיכות ומתקיים
$$
.([I]_C^B)^{-1}=[I]_B^C
$$

\end{proof}

\begin{remark}
בפרט, כאשר אחד הבסיסים הוא סטנדרטי (למשל
$C=E$)
קל לקבל את
$[I]_E^B$
בלי דירוג. נחשב את המטריצה ההופכית ונקבל
$$
.[I]_B^E=([I]_E^B)^{-1}
$$
\end{remark}

מה הקשר בין מטריצות מייצגות של אותה העתקה לינארית? לשם כך נועדו מטריצות המעבר.

\begin{proposition}\label{prop:rep-transition}
תהי
$T:V\to W$
ה"ל. יהיו 
$B_1,B_2$
בסיסים סדורים ל-
$V$,
ויהיו
$C_1,C_2$
בסיסים סדורים ל-
$W$. 
אז מתקיים
$$
.[T]_{C_2}^{B_2}=[I]_{C_2}^{C_1}[T]_{C_1}^{B_1}[I]_{B_1}^{B_2}
$$
\end{proposition}

\begin{proof}
לפי טענה
\ref{prop:rep-matrix}
מתקיים:
$$
\begin{aligned}
[T]_{C_2}^{B_2}&=[(\Id_W\circ T)\circ \Id_V]_{C_2}^{B_2}=[\Id_W\circ T]_{C_2}^{B_1}[\Id_V]_{B_1}^{B_2} \\
&=[\Id_W]_{C_2}^{C_1}[T]_{C_1}^{B_1}[\Id_V]_{B_1}^{B_2}=[I]_{C_2}^{C_1}[T]_{C_1}^{B_1}[I]_{B_1}^{B_2}
\end{aligned}
$$
\end{proof}

כמסקנה, נקבל מקרה פרטי שיהיה רלוונטי במיוחד בפרק הבא:

\begin{corollary}\label{cor:rep-similar}
תהי
$T:V\to V$
ויהיו
$B_1,B_2$
בסיסים סדורים ל-
$V$.
אז מתקיים
$$
.[T]_{B_2}^{B_2}=[I]_{B_2}^{B_1}[T]_{B_1}^{B_1}([I]_{B_2}^{B_1})^{-1}
$$
\end{corollary}

\begin{proof}
נשתמש בטענה
\ref{prop:rep-transition}
עם
$C_1=B_1,C_2=B_2$
ונקבל
$$
.[T]_{B_2}^{B_2}=[I]_{B_2}^{B_1}[T]_{B_1}^{B_1}[I]_{B_1}^{B_2}=[I]_{B_2}^{B_1}[T]_{B_1}^{B_1}([I]_{B_2}^{B_1})^{-1}
$$
השוויון האחרון נובע מטענה
\ref{prop:rep-inverse}.

\end{proof}

\begin{remark}
בתרגיל 
\ref{exer:similar-det}
כבר עמדנו על אחד הקשרים בין מטריצה ריבועית
$A\in\M_{n\times n}(\F)$
למטריצה
$PAP^{-1}$
כאשר 
$P\in\M_{n\times n}(\F)$
הפיכה. ראינו כי
$$
.\det(PAP^{-1})=\det(A)
$$
בפרק הבא נגדיר באופן מסודר את המושג של דמיון מטריצות. 
$PAP^{-1}$
נקראת דומה ל-
$A$
לפי הגדרה זו. במסקנה 
\ref{cor:rep-similar}
מדובר על
$$
.A=[T]_{B_1}^{B_1},\quad P=[I]_{B_2}^{B_1},\quad PAP^{-1}=[T]_{B_2}^{B_2}
$$
נבין בפרק הבא מה המשמעות המלאה של דמיון מטריצות חוץ משוויון בין הדטרמיננטות.
\end{remark}

\subsection{שחזור העתקה לינארית מתוך מטריצה מייצגת}
יהיו
$$
B=\Set{v_1,...,v_n},\quad C=\Set{w_1,...,w_m}
$$
בסיסים סדורים ל-
$V,W$
בהתאמה.
תהי
$T:V\to W$
הנתונה ע"י מטריצה מייצגת
$[T]_C^B$.
איך נמצא נוסחה מפורשת ל-
$T$?
העמודות של מטריצה זו הן בדיוק הוקטורים
$$
.[T(v_1)]_C,...,[T(v_n)]_C
$$
לכל
$1\leq j\leq n$,
אם
$$
,[T(v_j)]_C=\begin{pmatrix}a_{1j}\\ \vdots \\ a_{mj}\end{pmatrix}
$$
אז לפי ההגדרה של וקטור קוארדינטות בהכרח מתקיים
$$
.T(v_j)=a_{1j}w_1+...+a_{mj}w_m
$$
בעזרת הנתונים האלה אפשר למצוא נוסחה מפורשת ל-
$T$
לפי האלגוריתם של מציאת העתקה לינארית לפי בסיס
(תת-פרק
\ref{section:map-by-basis}).
אבל יש דרך אחרת שקשורה למטריצות מעבר. היא דומה אך עשויה להיות יותר נוחה ומסודרת, ויותר הגיוני להשתמש בה כאשר מלכתחילה הנתון קשור למטריצה מייצגת.

הרעיון: לחשב את את
$[T]_{E'}^E$
כאשר
$E$
הבסיס הסטנדרטי ל-
$V$,
ו-
$E'$
הבסיס הסטנדרטי ל-
$W$.
ניתן לעשות זאת לפי טענה
\ref{prop:rep-transition}
באופן הבא:
$$
[T]_{E'}^E=[I]_{E'}^C[T]_C^B[I]_B^E=[I]_{E'}^C[T]_C^B([I]_E^B)^{-1}
$$
נזכיר כי
$[T]_C^B$
נתונה, ואילו המטריצות
$$
[I]_{E'}^C, [I]_E^B
$$
מתקבלות באופן מיידי. לכן עיקר העבודה הוא החישוב של
$
,([I]_E^B)^{-1}
$
ולבסוף חישוב המכפלה של שלוש מטריצות. ברגע שנקבל את
$[T]_{E'}^E$,
המעבר לנוסחה מפורשת יהיה מיידי כי הבסיסים סטנדרטיים.

\begin{exercise}
יהיו 
$$
B=\Set{\begin{pmatrix}1\\2\end{pmatrix},\begin{pmatrix}3\\1\end{pmatrix}},\quad C=\Set{\begin{pmatrix}1\\-2\end{pmatrix},\begin{pmatrix}1\\3\end{pmatrix}}
$$
בסיסים סדורים ל-
$\R^2$.
מצאו ה"ל
$T:\R^2\to\R^2$
המקיימת
$$
.[T]_C^B=\begin{pmatrix}
2 & 5 \\
3 & 1
\end{pmatrix}
$$
\end{exercise}

\begin{solution}
נמצא מטריצות מעבר באופן מיידי:
$$
[I]_E^C=\begin{pmatrix}
1 & 1 \\
-2 & 3
\end{pmatrix},\quad [I]_E^B=\begin{pmatrix}
1 & 3 \\
2 & 1
\end{pmatrix}
$$

נחשב את
$[I]_B^E$
ע"י חישוב מטריצה הופכית. נעשה זאת לפי הנוסחה מטענה
\ref{prop:det-order-2}:
$$
[I]_B^E=([I]_E^B)^{-1}=\begin{pmatrix}
1 & 3 \\
2 & 1
\end{pmatrix}^{-1}=\frac{1}{1-6}\begin{pmatrix}
1 & -3 \\
-2 & 1
\end{pmatrix}=\begin{pmatrix}
-\frac{1}{5} & \frac{3}{5} \\
\frac{2}{5} & -\frac{1}{5}
\end{pmatrix}
$$

לפי טענה
\ref{prop:rep-transition}
נובע כי
$$
[T]_E^E=[I]_{E}^C[T]_C^B([I]_E^B)^{-1}=\begin{pmatrix}
1 & 1 \\
-2 & 3
\end{pmatrix}\begin{pmatrix}
2 & 5 \\
3 & 1
\end{pmatrix}\begin{pmatrix}
-\frac{1}{5} & \frac{3}{5} \\
\frac{2}{5} & -\frac{1}{5}
\end{pmatrix}=\begin{pmatrix}
\frac{7}{5} & \frac{9}{5} \\
-\frac{19}{5} & \frac{22}{5}
\end{pmatrix}
$$

לסיכום, נקבל את הנוסחה המפורשת של
$T$:
$$
T\begin{pmatrix}x\\y\end{pmatrix}=\left[T\begin{pmatrix}x\\y\end{pmatrix}\right]_E=[T]_E^E\left[\begin{pmatrix}x\\y\end{pmatrix}\right]_E=\begin{pmatrix}
\frac{7}{5} & \frac{9}{5} \\
-\frac{19}{5} & \frac{22}{5}
\end{pmatrix}\begin{pmatrix}x\\y\end{pmatrix}=\begin{pmatrix}\frac{7x+9y}{5} \\ \frac{22y-19x}{5}\end{pmatrix} 
$$
ניתן לבדוק (ליתר ביטחון) כי עבור הנוסחה לעיל אכן מתקיים
$$
.[T]_C^B=\begin{pmatrix}
2 & 5 \\
3 & 1
\end{pmatrix}
$$

\end{solution}

\subsection{גרעין ותמונה בהקשר של מטריצה מייצגת}

תהי
$A\in\M_{m\times n}(\F)$.
ראינו בטענה 
\ref{prop:matrix-map}
שמתקיים
$$
.\Ker(T_A)=\Null(A),\quad \Image(T_A)=\Col(A)
$$

זהו מקרה מיוחד של העתקה לינארית המוגדרת ע"י מטריצה באופן ישיר (הבסיסים המתאימים למטריצה המייצגת הם סטנדרטיים). מה ניתן לומר לגבי העתקה לינארית כללית בעזרת מטריצה מייצגת כלשהי שלה? ההכללה מופיעה בטענה הבאה:

\begin{proposition}\label{prop:ker-image-rep}
תהי
$T:V\to W$
ה"ל, ויהיו
$B,C$
בסיסים סדורים ל-
$V,W$
בהתאמה. אז מתקיים:
\begin{enumerate}[label=\alph*.]
\item 
לכל
$v\in V$
$$
.v\in\Ker(T) \iff [v]_B\in\Null([T]_C^B)
$$


\item 
לכל
$w\in W$
$$
.w\in\Image(T) \iff [w]_B\in\Col([T]_C^B)
$$


\end{enumerate}
\end{proposition}

\begin{proof}
נשתמש בטענה 
\ref{prop:rep-matrix}.

\begin{enumerate}[label=\alph*.]
\item 
יהי
$v\in V$.
מתקיים:
$$
\begin{aligned}
v\in\Ker(T) &\iff T(v)=0_W \iff [T(v)]_C=\underline{0} \iff [T]_C^B[v]_B=\underline{0} \\
&\iff [v]_B\in\Null([T]_C^B)
\end{aligned}
$$


\item 
יהי
$w\in W$.
נסמן
$B=\Set{v_1,...,v_n}$
ונקבל לפי טענה
\ref{prop:image-span}:
$$
\begin{aligned}
w\in\Image(T) &\iff w\in\Span(T(v_1),...,T(v_n)) \\
&\iff [w]_C\in \Span([T(v_1)]_C,...,[T(v_n)]_C)  \iff [w]_C\in \Col([T]_C^B))
\end{aligned}
$$

\end{enumerate}
\end{proof}

\begin{exercise}
בסימונים של הטענה לעיל, הוכיחו כי:
\begin{enumerate}[label=\alph*.]
\item 
$T$
חח"ע אם ורק אם
$\Null([T]_C^B)=\Set{\underline{0}}$.


\item 
$T$
על אם ורק אם
$\rank([T]_C^B)=m$.


\end{enumerate}
\end{exercise}

\begin{solution}
\begin{enumerate}[label=\alph*.]
\item 
לפי טענה
\ref{prop:injective},
$T$
חח"ע אם ורק אם 
$\Ker(T)=\Set{\underline{0}}$.
לפי טענה
\ref{prop:ker-image-rep},
התנאי האחרון שקול לתנאי
$\Null([T]_C^B)=\underline{0}$,
כי מתקיים
$$
.v=0_W \iff [v]_B=\underline{0}
$$


\item 
לפי טענה
\ref{prop:image-span},
$T$
על אם ורק אם
$\Span(T(v_1),...,T(v_n))=W$.
לפי טענה
\ref{prop:ker-image-rep},
התנאי האחרון שקול לתנאי
$\Col([T]_C^B)=\F^m$.
לבסוף, לפי משפט
\ref{thm:rank-nullity},
מתקיים
$$
.\dim(\Col([T]_C^B))=\rank([T]_C^B)
$$
לאור זאת ולפי לפי מסקנה 
\ref{cor:subspace-dim},
הדרגה שווה ל-
$m$
אם ורק אם
$\Col([T]_C^B)=\F^m$.
לסיכום,
$T$
על אם ורק אם 
$\rank([T]_C^B)=m$.


\end{enumerate}
\end{solution}

\section{איזומורפיזמים}

יש חשיבות לשילוב בין התכונה "חד-חד-ערכית" לתכונה "על". מכאן ההגדרה הבאה:

\begin{definition}
ה"ל
$T:V\to W$
נקראת איזומורפיזם אם היא גם חח"ע וגם על.
\end{definition}

כבר הוכחנו את משפט
\ref{thm:3-for-2-iso},
שהוא בעצם משפט "שלישי חינם" לאיזומורפיזמים. בפרט, המשפט קובע כי עבור איזומורפיזם 
$T:V\to W$
בהכרח מתקיים
$\dim(V)=\dim(W)$.
לחילופין, אם ידוע שיש שוויון מימדים, אז די להוכיח
כי $T$
היא חח"ע או על כדי להסיק שגם התכונה השנייה מתקיימת, ולכן 
$T$
היא איזומורפיזם.

\begin{example}
תהי
$T:M_{n\times n}(\F)\to M_{n\times n}(\F)$
העתקת השחלוף המוגדרת ע"י
$$
.T(A)=A^t
$$
נראה כי 
$T$
היא איזומורפיזם. לפי משפט "שלישי חינם", די להראות כי היא 
חח"ע שהרי מימד התחום שווה למימד הטווח (המרחבים עצמם שווים). אם כן:
$$
A\in\Ker(T) \iff A^t=0_{n\times n} \iff (A^t)^t=(0_{n\times n})^t \iff A=0_{n\times n}
$$
לכן
$\Ker(T)=\Set{0_{n\times n}}$
ו-
$T$
היא חח"ע. מכאן שהיא איזורמורפיזם.

לחילופין, אפשר להראות כי היא על בעזרת תכונה בסיסית של שחלוף, ולא הרבה מעבר לזה.
תהי
$B\in\M_{n\times n}(\F)$.
נפתור את המשוואה (כאשר הנעלם הוא המטריצה 
$A$)
הבאה:
$$
A^t=B
$$
נפעיל את העתקת השחלוף ונקבל
$$
,A=B^t
$$
ואכן מתקיים
$$
(B^t)^t=B
$$
כנדרש. לכן,
$T$
היא על כי בחרנו
$B$
שרירותית. אז שוב הראינו כי 
$T$
היא איזומורפיזם.
\end{example}

בדוגמה לעיל ראינו איזומורפיזם שהוא למעשה פעולה (במקרה זה על מטריצות) שיש לה פעולה הפוכה, כלומר בהינתן 
$T(A)$
ניתן לשחזר את 
$A$.
נרצה להבין שזהו אפיון כללי של איזומורפיזמים. ראשית, טענת עזר:

\begin{proposition}\label{prop:iso-by-basis}
תהי 
$T:V\to W$
ה"ל, ויהי
$B=\Set{v_1,...,v_n}$
בסיס ל-
$V$.
אז 
$\Set{T(v_1),...,T(v_n)}$
היא בסיס ל-
$W$
אם ורק אם
$T$
היא איזומורפיזם.
\end{proposition}

\begin{proof}
בכיוון אחד, נניח כי
$\Set{T(v_1),...,T(v_n)}$
היא בסיס ל-
$W$.
בפרט, מתקיים
$$
.\dim(W)=\dim(V)=n
$$
בנוסף, 
$T$
על כי מתקיים
$$
.\Image(T)=\Span(T(v_1),...,T(v_n))=W
$$
לכן, לפי משפט "שלישי חינם" לאיזומורפיזמים, נובע כי
$T$
היא איזומורפיזם.

בכיוון השני, נניח כי $T$
היא איזומורפיזם. בפרט, מתקיים
$$
.\dim(W)=\dim(V)=n
$$
כדי להראות כי 
$\Set{T(v_1),...,T(v_n)}$
היא בסיס ל-
$W$,
די להראות כי היא קבוצה פורשת. ובכן:
$$
\Span(T(v_1),...,T(v_n))=\Image(T)=W
$$
כלומר היא אכן קבוצה פורשת, ולכן היא בסיס לפי משפט "שלישי חינם" לבסיסים.

\end{proof}

\begin{definition}
ה"ל 
$T:V\to W$
נקראת הפיכה אם קיימת ה"ל
$T^{-1}:W\to V$
כך ש-
$$
.T\circ T^{-1}=\Id_W,\quad T^{-1}\circ T=\Id_V
$$
\end{definition}

$T^{-1}$
נקראת העתקה הופכית. המשמעות המעשית של ההגדרה היא הקשר הבא:
$$
T(v)=w \iff v=T^{-1}(w)
$$
או במילים: 
$v=T^{-1}(w)$
הוא הפתרון היחיד למשוואה
$T(v)=w$
עבור 
$w$
נתון.

% הדגמה או ציור

ההגדרה של ה"ל הפיכה אנלוגית לזו של מטריצה הפיכה. גם במקרה זה ניתן לוותר על אחת מהדרישות של ההגדרה, לפחות כאשר נתון שוויון מימדים בין התחום לטווח.

\begin{proposition}\label{prop:inverse-simplified}
יהיו
$T:V\to W$
ו-
$S:W\to V$
ה"ל כך ש-
$\dim(W)=\dim(V)$.
אז מתקיים
$$
.S\circ T=\Id_V \iff T\circ S=\Id_W
$$
\end{proposition}

\begin{proof}
נניח כי
$S\circ T=\Id_V$
ונוכיח כי
$T\circ S=\Id_W$
(הכיוון השני דומה מטעמי סימטריה).
לפי תרגיל
\ref{exer:composition},
$T$
חח"ע כי
$S\circ T$
חח"ע בתור העתקת הזהות.
לכן,
$T$
גם על לפי משפט "שלישי חינם" לאיזומורפיזמים. לבסוף, יהי
$w\in W$.
קיים
$v\in V$
כך ש-
$w=T(v)$,
ובהתאם נקבל
$$
.T\circ S(w)=T\circ S(T(v))=T(S\circ T(v))=T(\Id_V(v))=T(v)=w
$$
כלומר
$T\circ S=\Id_W$,
כנדרש.
\end{proof}

\begin{proposition}\label{prop:iso-invertible}
ה"ל
$T:V\to W$
היא איזומורפיזם אם ורק אם היא הפיכה.
\end{proposition}

\begin{proof}
בכיוון אחד, נניח כי
$T$
היא איזומורפיזם. בפרט,מתקיים
$\dim(V)=\dim(W)$.
יהי
$B=\Set{v_1,...,v_n}$
בסיס ל-
$V$.
לפי טענה 
\ref{prop:iso-by-basis},
נובע כי
$\Set{T(v_1),...,T(v_n)}$
היא בסיס ל-
$W$.
לכן, לפי משפט ההגדרה (משפט
\ref{thm:exist-unique})
קיימת ויחידה ה"ל
$S:W\to V$
המקיימת לכל
$1\leq j\leq n$
את הדרישה
$$
.S(T(v_j))=v_j
$$
באופן שקול, לכל
$1\leq j\leq n$
מתקיים
$$
,S\circ T(v_j)=\Id_V(v_j)
$$
ולכן
$$
S\circ T=\Id_V
$$
שוב לפי משפט ההגדרה. מטענה
\ref{prop:inverse-simplified}
נובע כי גם מתקיים
$$
.T\circ S=\Id_W
$$
לסיכום, קיבלנו כי
$T$
הפיכה ומתקיים
$S=T^{-1}$.

בכיוון השני, נניח כי
$T$
הפיכה. אז קיימת 
$T^{-1}:W\to V$
כך ש-
$$
.T^{-1}\circ T=\Id_V,\quad T\circ T^{-1}=\Id_W
$$
לפי תרגיל
\ref{exer:composition},
נובע מהשוויון 
$T^{-1}\circ T=\Id_V$
כי
$T$
חח"ע. מהשוויון השני נובע כי היא על (וגם ממשפט "שלישי חינם"). לסיכום, 
$T$
היא איזומורפיזם.

\end{proof}

בהינתן איזומורפיזם 
$T:V\to W$,
איך מוצאים את
$T^{-1}:W\to V$?
הוכחת המשפט לעיל מסבירה את הרעיון הכללי: לוקחים בסיס
$B=\Set{v_1,...,v_n}$
ל-
$V$
ומקבלים את הבסיס המתאים ל-
$W$:
$$
C=\Set{T(v_1),...,T(v_n)}
$$
ההעתקה ההופכית 
$T^{-1}$
היא ההעתקה הלינארית היחידה השולחת כל וקטור ב-
$C$
למקור שלו ב-
$B$.

שוב נרצה לראות גישה קצת שונה כדי שנוכל לעשות חישובים עם מטריצות, שיש לנו איתן יותר ניסיון מפרקים קודמים. התרגיל הבא הוא הכנה טובה:

\begin{exercise}
תהי
$T:\R^2\to\R^2$
המוגדרת ע"י
$$
.T\begin{pmatrix}x\\y\end{pmatrix}=\begin{pmatrix}
3x+2y \\
x+y
\end{pmatrix}
$$
הראו כי
$T$
היא איזומורפיזם ומצאו את
$T^{-1}$.
\end{exercise}

\begin{solution}
כאן
$$
,T=T_A
$$
כאשר
$$
.A=\begin{pmatrix}
3 & 2 \\
1 & 1
\end{pmatrix}
$$

לפי טענה 
\ref{prop:homeo},
לכל
$B\in\M_{2\times 2}(\R)$
מתקיים
$$
.T_A T_B=T_{AB}
$$
נראה בהמשך כי
$A$
הפיכה. בהנחה שזה נכון, נוכל להציב
$B=A^{-1}$
ולקבל
$$
.T_A T_{A^{-1}}=T_{I_2}=\Id_{\R^2}
$$
מכאן נסיק כי 
$T_A$
הפיכה, ולכן היא איזומורפיזם לפי טענה
\ref{prop:iso-invertible}. 
בנוסף, מתקיים
$$
.(T_{A})^{-1}=T_{A^{-1}}
$$

נסיים את הפתרון ע"י חישוב
$\det(A)$
ושימוש בנוסחה למטריצה הופכית מסדר
$2\times 2$:
$$
\det(A)=\begin{pmatrix}
3 & 2 \\
1 & 1
\end{pmatrix}=1\neq 0
$$
לכן זו מטריצה הפיכה, והמטריצה ההופכית נתונה ע"י
$$
.A^{-1}=\frac{1}{1}\begin{pmatrix}
1 & -2 \\
-1 & 3
\end{pmatrix}=\begin{pmatrix}
1 & -2 \\
-1 & 3
\end{pmatrix}
$$

לבסוף:
$$
T^{-1}\begin{pmatrix}x\\y\end{pmatrix}=T_A^{-1}\begin{pmatrix}x\\y\end{pmatrix}=\begin{pmatrix}
1 & -2 \\
-1 & 3
\end{pmatrix}\begin{pmatrix}x\\y\end{pmatrix}=\begin{pmatrix}x-2y\\-x+3y\end{pmatrix}
$$

ניתן לבדוק באופן ישיר (בלי מטריצות) כי אכן
$$
.T\circ T^{-1}=T^{-1}\circ T=\Id_{\R^2}
$$

\end{solution}

\begin{proposition}\label{prop:iso-by-matrix}
תהי
$T:V\to W$
ה"ל, ויהיו
$B,C$
בסיסים סדורים ל-
$V,W$
בהתאמה. אז 
$T$
היא איזומורפיזם אם ורק אם
$[T]_C^B$
הפיכה, ובמקרה זה מתקיים
$$
.[T^{-1}]_B^C=([T]_C^B)^{-1}
$$
\end{proposition}

\begin{proof}
בכיוון אחד, נניח כי 
$T$
היא איזומורפיזם. לכן, היא הפיכה לפי טענה 
\ref{prop:iso-invertible}.
פירוש הדבר שקיימת 
$T^{-1}$
כך ש-
$$
.T\circ T^{-1}=\Id_W
$$
נסמן
$n=\dim(V)=\dim(W)$.
לפי טענה
\ref{prop:rep-transition}
נובע כי
$$
.[T]_C^B[T^{-1}]_B^C=[T\circ T^{-1}]_C^C=[\Id_W]_C^C=I_{n}
$$
המטריצה המייצגת
$[T]_C^B$
היא מטריצה ריבועית מסדר 
$n\times n$,
ולכן היא הפיכה לפי החישוב לעיל וטענה 
\ref{prop:invertible}. 
בנוסף, קיבלנו כי
$$
.([T]_C^B)^{-1}=[T^{-1}]_B^C
$$

בכיוון השני, נניח כי
$T$
אינה איזומורפיזם ונראה כי
$[T]_C^B$
אינה הפיכה. ניתן להניח כי
$\dim(V)=\dim(W)$,
שכן אחרת המטריצה המייצגת אינה ריבועית ובפרט אינה הפיכה.
אם כן, ההנחה ש-
$T$
אינה איזומורפיזם פירושה שהיא אינה חח"ע (לפי משפט "שלישי חינם"). אז קיים
$0_V\neq v\in \Ker(T)$,
ולכן נובע מטענה
\ref{prop:ker-image-rep}
כי
$$
.\underline{0}\neq[v]_B\in \Null([T]_C^B)
$$
זה מראה כי
$[T]_C^B$
אינה הפיכה לפי משפט
\ref{thm:invertible}.
\end{proof}

\begin{exercise}
תהי
$A\in\M_{n\times n}(\F)$.
הוכיחו כי
$T_A:\F^n\to\F^m$
היא איזומורפיזם אם ורק אם 
$A$
הפיכה. מצאו נוסחה ל-
$T_A^{-1}$.

\end{exercise}

\begin{solution}\
זהו מקרה פרטי של טענה
\ref{prop:iso-by-matrix}
כאשר
$B,C$
הם הבסיס הסטנדרטי
$E$
ל-
$\F^n$.
לפי הטענה,
$T_A$
היא איזומורפיזם אם ורק אם 
$A=[T_A]_E^E$
הפיכה. נמצא נוסחה ל-
$T_A^{-1}$
באופן ישיר ע"י פתרון המשוואה הבאה עבור
$w\in\F^n$
שרירותי:
$$
Av=w
$$
הפתרון היחיד הוא 
$v=A^{-1}w$
ומתקיים
$$
.T_A(A^{-1}w)=w
$$
כלומר
$T_A$
הפיכה
(כידוע)
ומתקיים
$$
.T_A^{-1}(w)=A^{-1}w
$$
או באופן תמציתי:
$$
(T_A)^{-1}=T_{A^{-1}}
$$
\end{solution}

\begin{exercise}
\begin{enumerate}[label=\alph*.]
\item
תהי
$A\in M_{n\times n}(\F)$
הפיכה. נגדיר ה"ל
$S_A:\M_{n\times n}(\F)\to\M_{n\times n}(\F)$
ע"י
$$
.S_A(B)=AB
$$
הוכיחו כי היא איזומורפיזם. מצאו נוסחה ל-
$S_A^{-1}$.


\item
כעת נניח כי
$A$
אינה הפיכה. הוכיחו כי
$S_A$
אינה איזומורפיזם.

רמז: בשני הסעיפים אין צורך במטריצה מייצגת.


\end{enumerate}
\end{exercise}

\begin{solution}\
\begin{enumerate}[label=\alph*.]
\item
יש לציין שהמטריצה המייצגת אינה 
$A$,
אלא מטריצה אחרת (הקשורה אליה) מסדר 
$n^2\times n^2$.
במקום לחשב אותה הרבה יותר קל
להוכיח כי
$S_A$
הפיכה (ולכן איזומורפיזם) ע"י פתרון המשוואה הבאה:
$$
AB=C
$$
כאן הנעלם הוא המטריצה 
$B$.
נתון כי
$A$
הפיכה ולכן ניתן להכפיל את שני האגפים
ב-
$A^{-1}$
משמאל:
$$
B=A^{-1}C
$$
קיבלנו
$$
,S_A(A^{-1}C)=C
$$
ולכן
$S_A$
הפיכה ומתקיים
$$
.(S_A)^{-1}(C)=A^{-1}C
$$
או באופן תמציתי:
$$
(S_A)^{-1}=S_{A^{-1}}
$$


\item 
די להראות ש-
$S_A$
אינה על, או לחילופין שהיא אינה חח"ע. אין צורך בשתי הבדיקות, אך נעבור על שתיהן.

$S_A$
אינה על:
$A$
אינה הפיכה, ולכן אין פתרון למשוואה
$$
.AB=I_n
$$
זה מראה כי
$I_n\notin \Image(S_A)$,
כלומר 
$S_A$
אינה על. למעשה, לא קיימת אף מטריצה הפיכה ב-
$\Image(S_A)$.


$S_A$
אינה חח"ע: נתון כי
$A$
אינה הפיכה, ולכן קיים 
$\underline{0}\neq v\in\F^n$
כך ש-
$Av=\underline{0}$.
נסמן
$$
,B_v=\begin{pmatrix}
\begin{array}{c}
|\\[-0.4ex]
v\\[-0.4ex]
|
\end{array}
&
\begin{array}{c}
|\\[-0.4ex]
v\\[-0.4ex]
|
\end{array}
&
\cdots
&
\begin{array}{c}
|\\[-0.4ex]
v\\[-0.4ex]
|
\end{array}
\end{pmatrix}
$$
ונשים לב כי
$$
.S_A(B_v)=A\begin{pmatrix}
\begin{array}{c}
|\\[-0.4ex]
v\\[-0.4ex]
|
\end{array}
&
\begin{array}{c}
|\\[-0.4ex]
v\\[-0.4ex]
|
\end{array}
&
\cdots
&
\begin{array}{c}
|\\[-0.4ex]
v\\[-0.4ex]
|
\end{array}
\end{pmatrix}=\begin{pmatrix}
\begin{array}{c}
|\\[-0.4ex]
Av\\[-0.4ex]
|
\end{array}
&
\begin{array}{c}
|\\[-0.4ex]
Av\\[-0.4ex]
|
\end{array}
&
\cdots
&
\begin{array}{c}
|\\[-0.4ex]
Av\\[-0.4ex]
|
\end{array}
\end{pmatrix}=\0_{n\times n}
$$
קיבלנו כי
$0_{n\times n}\neq B_v\in\Ker(S_A)$,
ולכן
$\Ker(S_A)\neq \Set{0_{n\times n}}$.
זה מראה כי 
$S_A$
אינה חח"ע.


\end{enumerate}
\end{solution}
